Tools: Python, numpy, argparse, Monte Carlo simulation

Assignment 8 — Breast Cancer Prediction with Polygenic Risk Scores (assignment8/)

def confidence_interval(data, confidence=0.80):
    mean_val = np.mean(data)
    std_err = np.std(data) / np.sqrt(len(data))
    z = 1.28  # Z-score for 80% CI
    margin_of_error = z * std_err
    lower_bound = mean_val - margin_of_error
    upper_bound = mean_val + margin_of_error
    return lower_bound, upper_bound

def ttest_ind(data1, data2, equal_var=False):
    # Calculate the means
    mean1 = calculate_mean(data1)
    mean2 = calculate_mean(data2)
    
    # Calculate the variances
    var1 = calculate_variance(data1, mean1)
    var2 = calculate_variance(data2, mean2)
    
    # Sample sizes
    n1 = len(data1)
    n2 = len(data2)
    
    # Calculate the t-statistic
    t_stat = (mean1 - mean2) / math.sqrt((var1 / n1) + (var2 / n2))
    
    # Calculate degrees of freedom
    df = ((var1 / n1) + (var2 / n2)) ** 2 / (((var1 / n1) ** 2) / (n1 - 1) + ((var2 / n2) ** 2) / (n2 - 1))
    
    # Calculate two-tailed p-value
    p_value = 2 * (1 - t_dist_cdf(abs(t_stat), df))
    
    return t_stat, p_value

import numpy as np
import random

# Function to calculate the observed difference in means and medians
def observed_diff(data1, data2):
    mean_diff = np.mean(data1) - np.mean(data2)
    median_diff = np.median(data1) - np.median(data2)
    return mean_diff, median_diff

# Function to perform permutation test
def permutation_test(data1, data2, n_permutations=1000):
    combined_data = np.concatenate([data1, data2])
    n1 = len(data1)
    
    # Get observed differences in means and medians
    obs_mean_diff, obs_median_diff = observed_diff(data1, data2)
    
    # List to store permuted differences
    permuted_mean_diffs = []
    permuted_median_diffs = []
    
    for _ in range(n_permutations):
        np.random.shuffle(combined_data)
        permuted_data1 = combined_data[:n1]
        permuted_data2 = combined_data[n1:]
        
        # Calculate permuted differences
        perm_mean_diff, perm_median_diff = observed_diff(permuted_data1, permuted_data2)
        permuted_mean_diffs.append(perm_mean_diff)
        permuted_median_diffs.append(perm_median_diff)
    
    # Calculate p-values
    mean_p_value = np.mean(np.abs(permuted_mean_diffs) >= np.abs(obs_mean_diff))
    median_p_value = np.mean(np.abs(permuted_median_diffs) >= np.abs(obs_median_diff))
    
    return obs_mean_diff, obs_median_diff, mean_p_value, median_p_value

import argparse
import numpy as np

def simulate_coalescent(pop_size, sample_size, replicates):
    """
    Simulate the time to the eighth coalescent event in a coalescent model.
    
    Parameters:
        pop_size (int): The total population size.
        sample_size (int): The size of the sample for which the coalescent process is traced.
        replicates (int): The number of replicates for the simulation.
    
    Returns:
        (float, float): The mean and variance of the time to the eighth coalescent event.
    """
    coalescent_times = []

    for _ in range(replicates):
        num_lineages = sample_size
        gen_count = 0
        coalescent_events = 0
        
        while coalescent_events < 8 and num_lineages > 1:
            # Calculate coalescent probability for the current number of lineages
            coalescent_rate = num_lineages * (num_lineages - 1) / (2 * pop_size)
            time_to_coalescent = np.random.exponential(1 / coalescent_rate)
            gen_count += time_to_coalescent
            
            # Perform a coalescent event
            coalescent_events += 1
            num_lineages -= 1

        coalescent_times.append(gen_count)
    
    mean_generations = np.mean(coalescent_times)
    variance_generations = np.var(coalescent_times)
    
    return mean_generations, variance_generations

def main():
    parser = argparse.ArgumentParser(description="Simulate time to eighth coalescent event in a coalescent model.")
    parser.add_argument("--pop_size", type=int, required=True, help="The total population size.")
    parser.add_argument("--sample_size", type=int, required=True, help="The size of the sample to trace back.")
    parser.add_argument("--replicates", type=int, required=True, help="The number of simulation replicates.")
    
    args = parser.parse_args()
    
    mean_generations, variance_generations = simulate_coalescent(args.pop_size, args.sample_size, args.replicates)

import argparse
import numpy as np

def simulate_wright_fisher(allele_freq, pop_size, fitness):
    """
    Simulates allele frequency dynamics in a Wright-Fisher model with selection.
    
    Parameters:
        allele_freq (float): Initial allele frequency (between 0 and 1).
        pop_size (int): Population size.
        fitness (float): Relative fitness of the allele (1 for neutral, >1 for positive, <1 for negative).
    
    Returns:
        (int, bool): Number of generations until fixation or loss, and whether fixation occurred (True if fixed, False if lost).
    """
    gen_count = 0
    allele_count = int(allele_freq * pop_size)
    
    while 0 < allele_count < pop_size:
        gen_count += 1
        # Calculate probability of selecting allele A with selection
        prob_A = (allele_count / pop_size) * fitness / ((allele_count / pop_size) * fitness + (1 - allele_count / pop_size))
        # Simulate next generation's allele count
        allele_count = np.random.binomial(pop_size, prob_A)
    
    return gen_count, allele_count == pop_size

def main():
    # Parse command-line arguments
    parser = argparse.ArgumentParser(description="Wright-Fisher model with selection simulation")
    parser.add_argument("--allele_freq", type=float, required=True, help="Initial frequency of the allele (float between 0 and 1)")
    parser.add_argument("--pop_size", type=int, required=True, help="Population size (integer)")
    parser.add_argument("--fitness", type=float, required=True, help="Relative fitness of the allele (1 for neutral fitness, >1 for positive selection, <1 for negative selection)")
    parser.add_argument("--replicates", type=int, required=True, help="Number of Monte Carlo simulation replicates (integer)")
    args = parser.parse_args()

    # Run simulations
    fixation_times = []
    loss_times = []
    for _ in range(args.replicates):
        gen_count, fixed = simulate_wright_fisher(args.allele_freq, args.pop_size, args.fitness)
        if fixed:
            fixation_times.append(gen_count)
        else:
            loss_times.append(gen_count)
    
    # Calculate and print results
    if fixation_times:
        print(f"Allele was fixed in {np.mean(fixation_times):.2f} generations. Variance: {np.var(fixation_times):.2f}")
    if loss_times:
        print(f"Allele was lost in {np.mean(loss_times):.2f} generations. Variance: {np.var(loss_times):.2f}")

if __name__ == "__main__":
    main()

def load_population_schedule(pop_size_file):
    """
    Loads population size schedule from a TSV file.
    
    Parameters:
        pop_size_file (str): Path to the TSV file with generation and population size columns.
    
    Returns:
        List[Tuple[int, int]]: List of (generation, population size) tuples.
    """
    schedule = []
    with open(pop_size_file, 'r') as file:
        reader = csv.reader(file, delimiter='\t')
        next(reader)  # Skip the header
        for row in reader:
            generation = int(row[0])
            pop_size = int(row[1])

    Returns:

    for i, (start_gen, pop_size) in enumerate(schedule):
        if i == len(schedule) - 1 or gen < schedule[i + 1][0]:
            return pop_size

        allele_freq (float): Initial allele frequency (between 0 and 1).

        (int, bool): Number of generations until fixation or loss, and whether fixation occurred (True if fixed, False if lost).
    """

    pop_size = get_population_size(gen_count, schedule)
    allele_count = int(allele_freq * pop_size)

        gen_count += 1

        allele_freq = allele_count / pop_size
        if fitness > 0 and allele_freq > 0:

        else:
            prob_A = allele_freq  # Default to current frequency if fitness is neutral or allele freq is zero
        
        # Ensure prob_A is within [0, 1] to avoid NaNs or invalid probabilities
        prob_A = max(0, min(1, prob_A))

        allele_count = np.random.binomial(pop_size, prob_A)

def calculate_overlap(set_a, set_b):
    """Calculates the number of overlapping bases between two sets of ranges, chromosome-wise."""
    overlap = 0

    for chrom in set_a:
        if chrom in set_b:
            a_ranges = sorted(set_a[chrom], key=lambda x: x[0])
            b_ranges = sorted(set_b[chrom], key=lambda x: x[0])
            i, j = 0, 0

                start_a, stop_a = a_ranges[i]

                    overlap += max(0, min(stop_a, stop_b) - max(start_a, start_b))
                
                if stop_a < stop_b:
                    i += 1
                else:
                    j += 1
    return overlap

    # The matrix that generates the Fibonacci sequence
    fib_matrix = np.array([[1, 1], [1, 0]])

    # Solving for the coefficients in the linear combination of eigenvectors

    coeffs = np.linalg.solve(eigenvectors, start_vector)
    
    # Calculate nth term using eigenvalues raised to the power of (n-1)

                  coeffs[1] * (eigenvalues[1]**(n-1)) * eigenvectors[:, 1]
    
    # Return the integer part of the term

def nelder_mead(func, x0, alpha=1.0, gamma=2.0, rho=0.5, sigma=0.5, max_iter=100):
    n = len(x0)
    simplex = np.zeros((n + 1, n))
    simplex[0] = x0
    for i in range(1, n + 1):
        simplex[i] = x0.copy()
        simplex[i, i - 1] += 0.05  # Slightly perturb each point
    
    func_vals = np.array([func(s) for s in simplex])

    for _ in range(max_iter):
        # Order the simplex by function values
        order = np.argsort(func_vals)
        simplex = simplex[order]
        func_vals = func_vals[order]
        
        # Calculate the centroid of all but the worst point
        centroid = np.mean(simplex[:-1], axis=0)
        
        # Reflect the worst point through the centroid
        reflected = centroid + alpha * (centroid - simplex[-1])
        f_reflected = func(reflected)
        
        if func_vals[0] <= f_reflected < func_vals[-2]:
            simplex[-1] = reflected
            func_vals[-1] = f_reflected
        elif f_reflected < func_vals[0]:
            # Expand the reflection
            expanded = centroid + gamma * (reflected - centroid)
            f_expanded = func(expanded)
            if f_expanded < f_reflected:
                simplex[-1] = expanded
                func_vals[-1] = f_expanded
            else:
                simplex[-1] = reflected
                func_vals[-1] = f_reflected
        else:
            # Contract the simplex
            contracted = centroid + rho * (simplex[-1] - centroid)
            f_contracted = func(contracted)
            if f_contracted < func_vals[-1]:
                simplex[-1] = contracted
                func_vals[-1] = f_contracted
            else:
                # Shrink the simplex
                simplex[1:] = simplex[0] + sigma * (simplex[1:] - simplex[0])
                func_vals[1:] = np.array([func(s) for s in simplex[1:]])

    return simplex[0]  # Return the best point

# Viterbi algorithm to identify inbred regions
def viterbi_algorithm(genotypes, transition_probabilities):
    inbred_regions = {}
    
    for individual, genotype_data in genotypes.items():
        # Initialize Viterbi variables
        n = len(genotype_data)
        states = ["inbred", "outbred"]  # Define the hidden states: 'inbred' or 'outbred'
        V = np.zeros((n, 2))  # Viterbi matrix: rows are positions, columns are states
        backpointer = np.zeros((n, 2), dtype=int)  # To track the best path
        
        # Initial probabilities (usually uniform or based on priors)
        initial_probs = np.array([0.5, 0.5])
        
        # Calculate the initial state probabilities based on the first position
        first_genotype = genotype_data[0][1]
        emission_probs = np.array([
            GENOTYPE_LIKELIHOODS[first_genotype] if state == "inbred" else (1 - GENOTYPE_LIKELIHOODS[first_genotype])
            for state in states
        ])
        
        V[0] = initial_probs * emission_probs
        
        # Fill in the Viterbi matrix for all positions
        for t in range(1, n):
            curr_genotype = genotype_data[t][1]
            emission_probs = np.array([
                GENOTYPE_LIKELIHOODS[curr_genotype] if state == "inbred" else (1 - GENOTYPE_LIKELIHOODS[curr_genotype])
                for state in states
            ])
            
            for state in range(2):  # 'inbred' or 'outbred' state
                max_prob = -np.inf
                best_state = None
                for prev_state in range(2):
                    prob = V[t-1, prev_state] * transition_probabilities[prev_state, state] * emission_probs[state]
                    if prob > max_prob:
                        max_prob = prob
                        best_state = prev_state
                V[t, state] = max_prob
                backpointer[t, state] = best_state
        
        # Traceback to find the best path
        best_path = []
        best_last_state = np.argmax(V[n-1])
        for t in range(n-1, -1, -1):
            best_path.append(states[best_last_state])
            best_last_state = backpointer[t, best_last_state]
        best_path.reverse()
        
        # Record inbred regions for this individual
        inbred_regions[individual] = []
        start_pos = None
        for t, state in enumerate(best_path):
            if state == "inbred":
                if start_pos is None:

                if start_pos is not None:

# Function to calculate clustering error
def calculate_clustering_error(labels, true_clades):
    total_error = 0
    for cluster_id in np.unique(labels):
        # Get all the indices of samples in the current cluster
        cluster_indices = np.where(labels == cluster_id)[0]
        # Find the most common true label in this cluster
        majority_label = Counter(true_clades[cluster_indices]).most_common(1)[0][0]
        # Calculate the number of misclassified samples
        misclassified_count = np.sum(true_clades[cluster_indices] != majority_label)
        total_error += misclassified_count
    return total_error
