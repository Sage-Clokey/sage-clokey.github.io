README

C Systems Programming

Systems programming assignments in C from UC Santa Cruz CSE 13S (Computer Systems and C Programming).

Projects

Hello World (hello/)

Introduction to C: basic I/O, compilation, and the Unix toolchain.

Dice Game — Left Center Right (dice-game/)

Simulation of the LCR dice game using pseudo-random number generation. - Custom random number generator (cse13s_random.c) - Player names from array (names.h) - Game loop with dice rolling and chip passing logic

Hangman (hangman/)

Interactive word-guessing game in C. - Helper functions separated into hangman_helpers.c/.h - String manipulation, user input handling - ASCII art display of gallows state

Hex Dump (hex-dump/)

A xxd-style hex dump utility (xd.c). - Reads binary input and displays as hex + ASCII side by side - Similar to Unix xxd or od tools

RPN Calculator (rpn-calculator/)

Reverse Polish Notation calculator with custom math library. - Stack implementation (stack.c/.h) - Math functions (sin, cos, sqrt, etc.) (mathlib.c/.h) - Operator dispatch (operators.c/.h) - Main REPL loop (calc.c)

Hash vs Linked List Benchmarking (hash-benchmarking/)

Performance comparison of hash table vs linked list for dictionary lookup. - Hash table implementation (hash.c/.h) - Linked list implementation (ll.c/.h) - Dictionary word lookup benchmarks (bench1.c, bench2.c) - Abstract interface (dictwords.c/.h) with two concrete implementations

Graph Pathfinding (graph-pathfinding/)

Graph traversal and shortest path finding. - Graph ADT (graph.h) - Path representation (path.h) - Stack for DFS (stack.h) - Vertex management (vertices.h)

/*
 * Compute the absolute value of x.
 */
double Abs(double x) {
    if (x < 0) {
        return -x;
    } else {
        return x;
    }
}

/*
 * Compute the square root of x >= 0.
 */
double Sqrt(double x) {

    if (x < 0) {
        // Handle the case of negative input
        return (double) NAN; // Not a Number
    } else if (x == 0) {
        return 0.0;
    }

    // Initial guess (can be any positive number)
    double guess = x / 2.0;

    // Iteratively improve the guess using the Babylonian method
    while (Abs(guess * guess - x) > EPSILON) {
        guess = 0.5 * (guess + x / guess);
    }

    return guess;
}

/*
 * Compute the sine of x.
 */
double Sin(double x) {
    //take mode 2pi of x
    double xx = x;
    for (double n = 1; x + 2 * M_PI * n < 0; n++) {
        xx = x + 2 * M_PI * n;
    }
    x = fmod(xx, 2 * M_PI);
    // x = fmod(x, 2 * M_PI);

    double sum = 0;

    double factorial = 1;
    double power = x;
    double alternater = 1;
    double term = alternater * power / factorial;

    for (int n = 1; EPSILON < Abs(term); n++) {

        sum += term;

        factorial *= (2 * n + 1) * (2 * n);
        power *= x * x;
        alternater *= -1;

        term = alternater * power / factorial;
        //printf("term: %.20E , sum: %.20E, N: %d\n", term, sum, n);
    }
    return sum;
}

/*
 * Compute the cosine of x.
 */
double Cos(double x) {
    //take mode 2pi of x
    double xx = x;
    for (double n = 1; x + 2 * M_PI * n < 0; n++) {
        xx = x + 2 * M_PI * n;
    }
    x = fmod(xx, 2 * M_PI);

    double sum = 0;
    double term = 1;
    double factorial = 1;
    double power = 1;
    double alternater = 1;

    for (int n = 1; EPSILON < Abs(term); n++) {
        sum += term;

        factorial *= 2 * n * (2 * n - 1);
        power *= x * x;
        alternater *= -1;

        term = alternater * power / factorial;
    }
    return sum;
}

/*
 * Compute the tangent of x.
 *
 * NOTE:
 * Tan(M_PI / 2) will not be undefined for two reasons:
 * (1) the IEEE 754 double precision standard cannot perfectly represent
 *     pi in binary; and
 * (2) the Taylor-Maclaurin series used to approximate Cos(pi/2) will not
 *     converge to 0.
 *  Therefore, it will instead to converge to a very, very small
 *  number resulting in a very, very large value of Tan(pi/2).
 */
double Tan(double x) {
    return Sin(x) / cos(x);
}

typedef double (*binary_operator_fn)(double lhs, double rhs);

typedef double (*unary_operator_fn)(double x);

double operator_add(double lhs, double rhs) {
    double add = lhs + rhs;
    return add;
}

double operator_sub(double lhs, double rhs) {
    double subtract = lhs - rhs;
    return subtract;
}

double operator_mul(double lhs, double rhs) {
    double multiply = lhs * rhs;
    return multiply;
}

double operator_div(double lhs, double rhs) {
    double divide = lhs / rhs;
    return divide;
}

// return false if there are fewer than 2 items on the stack. otherwise, pop
// rhs, then pop lhs, then pass them into op, then push the result. and return
// true
bool apply_binary_operator(binary_operator_fn op) {
    if (stack_size < 2) {
        fprintf(stderr, ERROR_BINARY_OPERATOR);
        return false;
    }
    double lhs;
    double rhs;
    stack_pop(&rhs);
    stack_pop(&lhs);

    stack_push((op) (lhs, rhs));
    return true;
}

// return false if the stack is empty, otherwise pop an item, call op on it, and
// push the result
bool apply_unary_operator(unary_operator_fn op) {
    if (stack_size < 1) {
        fprintf(stderr, ERROR_UNARY_OPERATOR);
        return false;
    }
    double x;
    stack_pop(&x);
    stack_push((op) (x));
    return true;
}

#include "stack.h"

#include "mathlib.h"

#include <math.h>

#include <stdio.h>

double stack[STACK_CAPACITY];

int stack_size = 0;

bool stack_push(double item) {
    if (stack_size >= STACK_CAPACITY) {
        fprintf(stderr, ERROR_NO_SPACE, item);
        //stack_print();
        return false;
    }
    stack[stack_size] = item;
    stack_size += 1;
    return true;
}

bool stack_peek(double *item) {
    if (stack_size == 0) {
        return false;
    }
    *item = stack[stack_size - 1];
    return true;
}

bool stack_pop(double *item) {
    if (stack_size == 0) {
        return false;
    }
    *item = stack[stack_size - 1];
    stack_size -= 1;
    return true;
}

void stack_clear(void) {
    stack_size = 0;
}

void stack_print(void) {
    // make sure we don't print stack[0] when it is empty
    if (stack_size == 0) {
        return;
    }
    // print the first element with 10 decimal places
    printf("%.10f", stack[0]);
    // print the remaining elements (if any), with a space before each one
    for (int i = 1; i < stack_size; i++) {
        printf(" %.10f", stack[i]);
    }
    printf("\n");
}

            }
        }
        total_bytes_read += read_result; // Update total bytes read
    }
    *bytes_read = total_bytes_read; // Update bytes_read
    return true; // Successfully read BUFFER_SIZE bytes
}

Graph *graph_create(uint32_t vertices, bool directed) {
    Graph *g = calloc(1, sizeof(Graph));
    g->vertices = vertices;
    g->directed = directed;
    // use calloc to initialize everything with zeroes
    g->visited = calloc(vertices, sizeof(bool));
    g->names = calloc(vertices, sizeof(char *));

struct stack;
typedef struct stack {
    uint32_t capacity;
    uint32_t top;
    uint32_t *items;
} Stack;

Stack *stack_create(uint32_t capacity) {
    Stack *stack = malloc(sizeof(Stack));
    stack->items = calloc(sizeof(uint32_t), capacity);
    stack->top = 0;
    stack->capacity = capacity;
    return stack;
}

void stack_free(Stack **sp) {
    if (*sp) {
        free((*sp)->items);
        free(*sp);
        *sp = NULL;
    }
}

bool stack_push(Stack *s, uint32_t val) {
    if (s->capacity <= s->top) {
        return false;
    }
    s->items[s->top++] = val;
    return true;
}

bool stack_pop(Stack *s, uint32_t *val) {
    if (stack_empty(s)) {
        return false;
    }
    *val = s->items[s->top - 1];
    s->top -= 1;
    return true;
}

bool stack_peek(const Stack *s, uint32_t *val) {
    if (s->top == 0) {
        return false;
    }
    *val = s->items[s->top - 1];
    return true;
}

bool stack_empty(const Stack *s) {
    return s->top == 0;
}

bool stack_full(const Stack *s) {
    return s->top >= s->capacity;
}

uint32_t stack_size(const Stack *s) {
    return s->top;
}

void stack_copy(Stack *dst, const Stack *src) {

        dst->items = realloc(dst->items, src->capacity * sizeof(uint32_t));

    }

/*
* File:     cse13s_random.c
*
* Author:   Dr. Kerry Veenstra
*
* Purpose:  Generate a sequence of 1,000,000 pseudo-random numbers whose
*           values range between 0 and 999999.
*
*           We use this pseudo-random number generator for CSE 13S because
*           we want to be sure that all students' programs get the same
*           sequence of numbers.  Orinarily we would use random(), but we
*           know that random() can return a different sequence of numbers on
*           different systems.
*
* History:  A scaled version of this algorithm first was used in user
*           programs of early Hewlett-Packard programmable calculators.
*
*           The calculators used base-10 arithmetic (not binary).  They
*           could represent fractional values like 0.211327 exactly.
*
*           The calculator equation for computing the next pseudo-random
*           value of the sequence is below.  The frac() function zeroes
*           out the digits of the number to the left of the decimal point.
*
*               X_next = frac(9821 * X + 0.211327)
*
*           Scaling everything by 1000000 converts all values into
*           integers, and then the frac() function becomes the modulus
*           operator.
*
*               X_next = (9821 * X + 211327) % 1000000;
*
*           The value 9821 * X could be as large as 9821 * 999999 =
*           9,820,990,179, which is too large for a 32-bit variable, and so
*           we need to use a 64-bit variable.  (A 32-bit unsigned int can
*           store a value as large as 2^32 - 1 = 4,294,967,295.)
*
*           One can test this function by calling it 1000000 times and
*           then confirming that it generates 1000000 different values.
*/

/*
* DO NOT EDIT THIS FILE
*/

#include "cse13s_random.h"

#include <inttypes.h>

static uint64_t current_random = 0;

void cse13s_random_seed(unsigned seed) {
    current_random = (uint64_t) seed % 1000000;
}

int cse13s_random(void) {
    current_random = (9821 * current_random + 211327) % 1000000;
    return (int) current_random;
}

#define TABLE_SIZE 100 // Define the size of the hash table

typedef struct Node {
    char *key;
    int value;
    struct Node *next;
} Node;

typedef struct Hashtable {
    Node **buckets;
    int size;
} Hashtable;

unsigned int hash_function(const char *key) {
    unsigned long int hash = 0;
    int i = 0;
    while (hash < ULONG_MAX && i < strlen(key)) {
        hash = hash << 8;
        hash += key[i];
        i++;
    }
    return hash % TABLE_SIZE;
}

Hashtable *hash_create(void) {
    Hashtable *table = malloc(sizeof(Hashtable));
    table->buckets = malloc(sizeof(Node *) * TABLE_SIZE);
    table->size = TABLE_SIZE;
    for (int i = 0; i < TABLE_SIZE; i++) {
        table->buckets[i] = NULL;
    }
    return table;
}

bool hash_put(Hashtable *table, char *key, int val) {
    unsigned int index = hash_function(key);
    Node *new_node = malloc(sizeof(Node));
    if (!new_node)
        return false;

    new_node->key = strdup(key);
    new_node->value = val;
    new_node->next = table->buckets[index];
    table->buckets[index] = new_node;

    return true;
}

int *hash_get(Hashtable *table, char *key) {
    unsigned int index = hash_function(key);
    Node *node = table->buckets[index];
    while (node != NULL) {
        if (strcmp(node->key, key) == 0) {
            return &node->value;
        }
        node = node->next;
    }
    return NULL;
}

void hash_destroy(Hashtable **table) {
    if (table == NULL || *table == NULL)
        return;

    for (int i = 0; i < (*table)->size; i++) {
        Node *node = (*table)->buckets[i];
        while (node != NULL) {
            Node *temp = node;
            node = node->next;
            free(temp->key);
            free(temp);
        }
    }
    free((*table)->buckets);
    free(*table);
    *table = NULL;
}

        return true;
    }
}
