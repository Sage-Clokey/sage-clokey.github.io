class SBIRDocTemplate(BaseDocTemplate):
    def __init__(self, filename, section_title="", **kw):
        super().__init__(filename, **kw)
        self.section_title = section_title
        self._page_counter = [0]
        frame = Frame(
            self.leftMargin, self.bottomMargin,
            self.width, self.height,
            id="normal"
        )
        pt = PageTemplate(id="main", frames=[frame],
                          onPage=self._draw_chrome)
        self.addPageTemplates([pt])

    def _draw_chrome(self, c, doc):
        pg = doc.page
        # top rule
        c.setStrokeColor(DARK_GREEN)
        c.setLineWidth(1.5)
        c.line(doc.leftMargin, H - doc.topMargin + 6,
               W - doc.rightMargin, H - doc.topMargin + 6)
        # running head
        c.setFillColor(DARK_GREEN)
        c.setFont("Helvetica-Bold", 7.5)
        c.drawString(doc.leftMargin, H - doc.topMargin + 10,
                     "PI: Clokey, Sage  |  Spiral Steward, Inc.  |  SpiralCAD")
        c.setFont("Helvetica", 7.5)
        c.setFillColor(MID_GREEN)
        c.drawRightString(W - doc.rightMargin, H - doc.topMargin + 10,
                          doc.section_title)
        # bottom rule + page number
        c.setStrokeColor(LIGHT_GREEN)
        c.setLineWidth(0.8)
        c.line(doc.leftMargin, doc.bottomMargin - 8,
               W - doc.rightMargin, doc.bottomMargin - 8)
        c.setFillColor(GRAY)
        c.setFont("Helvetica", 7.5)
        c.drawCentredString(W / 2, doc.bottomMargin - 18, str(pg))

def H1(text): return Paragraph(f"  {text}", ST["section_h1"])
def H2(text): return Paragraph(text, ST["section_h2"])
def H3(text): return Paragraph(text, ST["section_h3"])
def B(text):  return Paragraph(text, ST["body"])
def BI(text): return Paragraph(text, ST["body_indent"])
def DL(text): return Paragraph(f"<i>Deliverable: {text}</i>", ST["deliverable"])
def SP(n=1):  return Spacer(1, n * 5)
def HR():     return HRFlowable(width="100%", thickness=0.8, color=RULE_COLOR,
                                 spaceAfter=4, spaceBefore=4)
def AIM(text): return Paragraph(text, ST["aim_label"])
def REF(text): return Paragraph(text, ST["ref"])
def CAP(text): return Paragraph(text, ST["caption"])

def BULLET(items):
    return ListFlowable(
        [ListItem(Paragraph(t, ST["bullet"]), bulletText="•") for t in items],
        bulletType="bullet", leftIndent=6, spaceAfter=2,
    )

    def __init__(self, path, section="", **kw):
        super().__init__(path, **kw)
        self.section = section
        frame = Frame(self.leftMargin, self.bottomMargin,
                      self.width, self.height, id="main")
        pt = PageTemplate(id="main", frames=[frame], onPage=self._chrome)
        self.addPageTemplates([pt])

    def _chrome(self, c, doc):
        pg = doc.page
        c.setStrokeColor(FOREST); c.setLineWidth(1.5)
        c.line(doc.leftMargin, H - doc.topMargin + 6,
               W - doc.rightMargin, H - doc.topMargin + 6)
        c.setFillColor(FOREST); c.setFont("Helvetica-Bold", 7.5)
        c.drawString(doc.leftMargin, H - doc.topMargin + 10,
                     "PI: Clokey, Sage  |  Spiral Steward, Inc.  |  LivingWorks")
        c.setFillColor(BARK); c.setFont("Helvetica", 7.5)
        c.drawRightString(W - doc.rightMargin, H - doc.topMargin + 10,
                          self.section)
        c.setStrokeColor(MOSS); c.setLineWidth(0.8)
        c.line(doc.leftMargin, doc.bottomMargin - 8,
               W - doc.rightMargin, doc.bottomMargin - 8)
        c.setFillColor(GRAY); c.setFont("Helvetica", 7.5)
        c.drawCentredString(W/2, doc.bottomMargin - 18, str(pg))

def wrap(text, maxch=55):
    words = text.split()
    lines, line = [], ""
    for w in words:
        if len(line)+len(w)+1 <= maxch:
            line = (line+" "+w).lstrip()
        else:
            if line: lines.append(line)
            line = w
    if line: lines.append(line)
    return lines

def slide_text(c, x, y, label, body, lh=13, maxch=50):
    c.setFillColor(FOREST); c.setFont("Helvetica-Bold", 8.5)
    c.drawString(x, y, label); y -= 14
    c.setFillColor(TEXT); c.setFont("Helvetica", 8.5)
    for ln in wrap(body, maxch):
        c.drawString(x, y, ln); y -= lh
    return y - 4

def slide_box(c, x, y, w, h, lines):
    c.setFillColor(FOREST)
    c.rect(x, y, w, h, fill=1, stroke=0)
    cy = y + h - 0.28*inch
    c.setFillColor(GOLD); c.setFont("Helvetica-Bold", 8)
    c.drawString(x+0.13*inch, cy, lines[0]); cy -= 14
    c.setFont("Helvetica", 8.5)
    for ln in lines[1:]:
        if ln == "---":
            c.setFillColor(GOLD); c.setFont("Helvetica-Bold", 8)
            cy -= 4
        elif ln.startswith("*"):
            c.setFillColor(colors.white); c.setFont("Helvetica", 8.5)
            c.drawString(x+0.13*inch, cy, ln[1:])
            cy -= 13
        else:
            c.setFillColor(colors.white); c.setFont("Helvetica", 8.5)

def mtext(c, x, y, txt, font, size, color, max_width, line_height, align="left"):
    """Multi-line text with word wrap."""
    c.setFillColor(color)
    c.setFont(font, size)
    words = txt.split()
    lines, line = [], ""
    for w in words:
        test = (line + " " + w).strip()
        if c.stringWidth(test, font, size) <= max_width:
            line = test
        else:
            if line: lines.append(line)
            line = w
    if line: lines.append(line)
    for ln in lines:
        c.setFont(font, size)
        if align == "center":
            c.drawCentredString(x, y, ln)
        elif align == "right":
            c.drawRightString(x, y, ln)
        else:
            c.drawString(x, y, ln)
        y -= line_height
    return y

def draw_spiral(c, cx, cy, r, color, lw=1.2, turns=2.2):
    """Approximate Fibonacci spiral using arc segments."""
    import math
    c.setStrokeColor(color)
    c.setLineWidth(lw)
    p = c.beginPath()
    theta = 0
    steps = 180
    for i in range(steps + 1):
        theta = (turns * 2 * math.pi / steps) * i
        radius = r * (theta / (turns * 2 * math.pi) + 0.05)
        x = cx + radius * math.cos(theta)
        y = cy + radius * math.sin(theta)
        if i == 0:
            p.moveTo(x, y)
        else:
            p.lineTo(x, y)
    c.drawPath(p, stroke=1, fill=0)

def draw_tree(c, x, y, length, angle, depth, color, lw):

def wrap_text(text, max_chars=90):
    """Simple word-wrap returning list of lines."""
    words = text.split()
    lines, line = [], ""
    for w in words:
        if len(line) + len(w) + 1 <= max_chars:
            line = (line + " " + w).lstrip()
        else:
            if line:
                lines.append(line)
            line = w
    if line:
        lines.append(line)
    return lines

def draw_header(c, aim_num, aim_title):

def draw_body_text(c, x, y, text, font="Helvetica", size=9, color=TEXT_BLACK,
                   max_chars=72, line_h=14, max_w=None):
    """Draw wrapped paragraph, return final y."""
    c.setFillColor(color)
    c.setFont(font, size)
    lines = wrap_text(text, max_chars)
    for ln in lines:
        c.drawString(x, y, ln)
        y -= line_h
    return y

def draw_box(c, x, y, w, h, bg, border=None):
    c.setFillColor(bg)
    if border:
        c.setStrokeColor(border)
        c.setLineWidth(1)
        c.rect(x, y, w, h, fill=1, stroke=1)
    else:
        c.rect(x, y, w, h, fill=1, stroke=0)

    c.setFillColor(color)
    c.setFont(font, size)
    c.drawString(x, y, text)
