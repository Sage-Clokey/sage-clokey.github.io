orf-finder

6-frame ORF finder for DNA sequences — searches forward and reverse complement strands with configurable codons and minimum length

README

ORF Finder

A Python tool for finding Open Reading Frames (ORFs) in DNA sequences using all 6 reading frames (3 forward + 3 reverse complement).

Features

  • Searches all 6 reading frames simultaneously
  • Generates reverse complement strand automatically
  • Configurable start codons (default: ATG)
  • Configurable stop codons (default: TAG, TAA, TGA)
  • Minimum ORF length filter
  • Handles dangling ends (ORFs that extend to sequence boundaries)
  • Accepts FASTA format input
  • Command-line interface with argparse

Usage

python mainOrfs.py --minGene 100 inputFile.fa

Options

Flag Default Description
--minGene 100 Minimum ORF length in nucleotides
--start ATG Start codon(s)
--stop TAG TAA TGA Stop codon(s)
--longestGene False Report only the longest ORF per position

Files

  • findOrfs.py — OrfFinder class with 6-frame search logic
  • mainOrfs.py — Entry point, argparse CLI, output formatting
  • sequenceAnalysis.py — NucParams, ProteinParam, FastAreader helper classes
  • tass2.fa — Sample FASTA input (chromosome data)
  • lab5test.fa — Small test FASTA

Output Format

+1 start..stop length
-2 start..stop length

Frame sign indicates strand (+/-), number indicates reading frame offset.

import os
from sequenceAnalysis import FastAreader, NucParams, ProteinParam
class FindOrfs():
    """
    Finds Open Reading Frames (ORFs) in a DNA sequence.

    Attributes:
        sequence (str): The input DNA sequence.
        startCodon (tuple): Tuple containing start codons. Default is ('ATG',).
        stopCodon (tuple): Tuple containing stop codons. Default is ('TAG', 'TAA', 'TGA').
        biggestOrfOnly (bool): Flag to indicate whether to output only the longest gene found. Default is False.
        minOrf (int): Minimum gene length to include. Default is 1.
        dangle (bool): Flag to indicate whether to output the dangling bit. Default is True.
        revSequence (str): The reverse complement of the input DNA sequence.
        geneCandidates (list): List containing gene candidate indexes, start, stop, length, and frame.
        sequenceLength (int): Length of the input DNA sequence.
        geneSequences (list): List containing gene sequences.
    """
    def __init__(self, sequence,startCodon = ('ATG'), stopCodon = ('TAG','TAA','TGA'),biggestOrfOnly = False, minOrf = 1,dangle = True, showGenes = False):
        """
        Initializes FindOrfs with the provided parameters.

        Args:
            sequence (str): The input DNA sequence.
            startCodon (tuple): Tuple containing start codons. Default is ('ATG',).
            stopCodon (tuple): Tuple containing stop codons. Default is ('TAG', 'TAA', 'TGA').
            biggestOrfOnly (bool): Flag to indicate whether to output only the longest gene found. Default is False.
            minOrf (int): Minimum gene length to include. Default is 1.
            dangle (bool): Flag to indicate whether to output the dangling bit. Default is True.
            showGenes (bool): Flag to indicate whether to show gene sequences. Default is False.
        """
        self.showGenes = showGenes
        self.dangle = dangle
        self.sequence = sequence
        self.revSequence = self.reverseSequence()
        self.startCodon = startCodon
        self.stopCodon = stopCodon
        self.biggestOrfOnly = biggestOrfOnly
        self.minOrf = minOrf
        self.geneCandidates = self.findGenes()
        self.sequenceLength = len(self.sequence)
        self.geneSequences = self.findGeneSequences()

    def reverseSequence(self):
        """
        Generates the reverse complement of the input DNA sequence.

        Returns:
            str: The reverse complement of the input DNA sequence.
        """
        complement = {'A': 'T', 'C': 'G', 'G': 'C', 'T': 'A'}
        reverse_sequence = ''.join([complement[base] for base in reversed(self.sequence)])
        return reverse_sequence
    
    def getGenes(self):
        return self.geneCandidates
    def getGeneSequences(self):
        return self.geneSequences
    
    def findGeneSequences(self):
        """
        Finds gene sequences.

        Returns:
            list: List containing gene sequences.
        """
        geneSequences = []
        for gene in self.geneCandidates:
            if gene[0] > 0:
                geneSequences.append(self.sequence[gene[1]-1:gene[2]])
                gene.append(self.sequence[gene[1]-1:gene[2]])
            else:
                #convert to reverse strand index
                start = (len(self.revSequence) - (gene[2]))

                stop = (len(self.revSequence) - gene[1]) + 1
                geneSequences.append(self.revSequence[start:stop])
                gene.append(self.revSequence[start:stop])
        return geneSequences

    def findGenes(self):
        """
        Finds Open Reading Frames (ORFs) in the DNA sequence.

        Returns:
            list: List containing gene candidate indexes, start, stop, length, and frame.
        """
        startCodonIndex = []
        geneCandidates = []
        

        #forward sequence
        for frame in range(3):
            startCodonIndex = []
            
            # Generate possible start codon positions for the current frame
            frameStarts = list(range(frame,  len(self.sequence) -2 , 3))
            
            for i in range(frame, len(self.sequence) - 2, 3):
                codon = self.sequence[i:i + 3]
                # beginning dangle
                
                if (codon in self.startCodon) or (self.dangle and i == frame):
                    startCodonIndex.append(i)
                
                elif codon in self.stopCodon or (self.dangle and i == frameStarts[-1]):
                    
                    if self.biggestOrfOnly == True:
                        biggestStart = float ('inf')
                        for start in startCodonIndex:
                            if start < biggestStart:
                                biggestStart = start
                        # Calculate gene length and append gene information if it meets minimum length criteria
                        geneCandidate= [frame + 1, biggestStart + 1, i + 3, i-biggestStart + 3]
                        if geneCandidate[3] >= self.minOrf:
                            geneCandidates.append(geneCandidate)
                    else:
                        for start in startCodonIndex:
                            # frame ,  start codon, sStop Codon , length      indexes for top strand
                            geneCandidate = [(frame + 1), start + 1, i + 3, i-start + 3]
                            if geneCandidate[3] >= self.minOrf:
                                geneCandidates.append(geneCandidate)
                    startCodonIndex = []
            

                       
                       
        #reverse sequence
        for frame in range(3):
            startCodonIndex = []

            # Generate possible start codon positions for the current frame on the reverse sequence
            frameStarts = list(range(frame, len(self.revSequence) - 2, 3))

            
            for i in range(frame,  len(self.revSequence) - 2, 3):
                codon = self.revSequence[i:i + 3]
                # beginning dangle
                
                if (codon in self.startCodon) or (self.dangle and i == frame):
                    startCodonIndex.append(i)
                
                elif codon in self.stopCodon or (self.dangle and i == frameStarts[-1]):
                    if self.biggestOrfOnly == True:
                        biggestStart = float ('inf')
                        for start in startCodonIndex:
                            if start < biggestStart:
                                biggestStart = start
                            # frame ,    stop codon , start codon,  length      indexes for top strand
                        geneCandidate = [-(frame + 1),   len(self.revSequence) - (i + 2), len(self.revSequence) - (biggestStart), (i-biggestStart) + 3]
                        if geneCandidate[3] >= self.minOrf:
                            geneCandidates.append(geneCandidate)
                    else:
                        for start in startCodonIndex:
                            geneCandidate= [-(frame + 1), len(self.revSequence) - (i + 2), len(self.revSequence) - (start), (i-start) + 3]
                            if geneCandidate[3] >= self.minOrf:
                                geneCandidates.append(geneCandidate)        
                    startCodonIndex = []
                
        #sort geneCandidates by length
        geneCandidates = sorted(geneCandidates, key=lambda gene: (-gene[3], gene[1]))  
        return geneCandidates
    
    def printGeneCandidates(self):
        """
        Prints information about the gene candidates.

        This method prints details such as start codon, stop codon, length, and frame of each gene candidate.
        """
        geneCandidatesTP = self.getGenes()
        #inf = [self.startCodon,self.stopCodon,self.biggestOrfOnly,self.minOrf]
        #print(f"{inf[0]:>15}{inf[1]:>15}{inf[2]:>15}{inf[3]:>15}")
        inf = [str(self.startCodon),str(self.stopCodon)]
        print("biggest Orf: " + str(self.biggestOrfOnly ))
        print("minimum Orf: " + str(self.minOrf))
        print("dangly: " + str(self.dangle))
        print("show Genes: " + str(self.showGenes))
        label = ["start Codon","Stop Codon"]
        print(f"{label[0]:>15}{label[1]:>30}")
        print(f"{inf[0]:>15}{inf[1]:>30}")
        info = ["Frame", "left Codon","Right Codon","Length"]
        print(f"{info[0]:>15}{info[1]:>15}{info[2]:>15}{info[3]:>15}")
        for gene in geneCandidatesTP:
            #print('gene: ', end=' ')
            for index, value in enumerate(gene):
                if index == 4:  # Check if it's the fifth value (index starts from 0)
                    if self.showGenes == True:
                        print(f'               {value:<15}', end='')  # Print with specific alignment
                else:
                    print(f'{value:>15}', end='')  # Print with default alignment

            print()
        print('sequence length: ' + str(self.sequenceLength))
        print('revsequence length: ' + str(len(self.revSequence)))
    

import os
import argparse


def main():
    """
    Entry point for the script.

    Parses command-line arguments to find ORFs in DNA sequences provided in a FASTA file.

    Command-line Arguments:
        filename (str): Path to the input FASTA file.
        --longestGene: Flag to indicate whether to output only the longest gene found.
        --start (str): Start codons. Default is 'ATG'.
        --stop (str): Stop codons. Default is 'TAG TAA TGA'.
        --minGene (int): Minimum gene length to include. Default is 1.
        --Nodangly: Flag to indicate whether to output the dangling bit. Default is True.
    """
    parser = argparse.ArgumentParser(description='Find Open Reading Frames (ORFs) in DNA sequences.')
    parser.add_argument('filename', metavar='filename', type=str, help='Path to the input FASTA file')
    parser.add_argument('--longestGene', action='store_true', help='Only output the longest gene found')
    parser.add_argument('--start', nargs='+', type=str, default=['ATG'], help='Start codons (default: ATG)')
    parser.add_argument('--stop', nargs='+', type=str, default=['TAG', 'TAA', 'TGA'], help='Stop codons (default: TAG TAA TGA)')
    parser.add_argument('--minGene', type=int, default=1, help='Minimum gene length to include (default: 1)')
    parser.add_argument('--Nodangly', action='store_false', help='dont output the dangling bit')
    parser.add_argument('--showGenes', action='store_true', help='show the gene sequences')

    
    args = parser.parse_args()

    print(f"\n{args.filename} analysis\n")

    myReader = FastAreader(fname=args.filename)
    for head, seq in myReader.readFasta():
        sequence = FindOrfs(seq, startCodon=args.start, stopCodon=args.stop, minOrf=args.minGene, biggestOrfOnly=args.longestGene, dangle=args.Nodangly, showGenes = args.showGenes)
        print(head)
        sequence.printGeneCandidates()

if __name__ == "__main__":
    main()
