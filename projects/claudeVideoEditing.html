"""
TikTok Edit Tool
Usage: edit the CLIPS list below, then run: python3 edit.py
"""

from moviepy import VideoFileClip, concatenate_videoclips, TextClip, CompositeVideoClip, AudioFileClip
import os

# ─────────────────────────────────────────────
# CONFIGURE YOUR EDIT HERE
# ─────────────────────────────────────────────

OUTPUT_FILE = "output.mp4"
TIKTOK_SIZE = (1080, 1920)  # vertical 9:16

# Each clip: (filename, start_seconds, end_seconds)
# Example: ("clip1.mp4", 10, 25) = from 0:10 to 0:25
CLIPS = [
    ("clip1.mp4", 0, 10),
    ("clip2.mp4", 5, 20),
]

# Optional: text overlays per clip — set to None to skip
# Each entry: (text, font_size, color, position, start_sec, end_sec)
# position can be "center", "bottom", "top", or (x, y)
TEXT_OVERLAYS = [
    # ("Fire is life, not just destruction.", 50, "white", "bottom", 0, 5),
]

# Optional: background music — set to None to skip
MUSIC_FILE = None  # e.g. "music.mp3"
MUSIC_VOLUME = 0.3  # 0.0 to 1.0

# ─────────────────────────────────────────────


def crop_to_tiktok(clip, target_size=TIKTOK_SIZE):
    """Center-crop clip to 9:16 vertical format."""
    target_w, target_h = target_size
    target_ratio = target_w / target_h

    orig_w, orig_h = clip.size
    orig_ratio = orig_w / orig_h

    if orig_ratio > target_ratio:
        # too wide — crop sides
        new_w = int(orig_h * target_ratio)
        x1 = (orig_w - new_w) // 2
        clip = clip.cropped(x1=x1, width=new_w)
    else:
        # too tall — crop top/bottom
        new_h = int(orig_w / target_ratio)
        y1 = (orig_h - new_h) // 2
        clip = clip.cropped(y1=y1, height=new_h)

    return clip.resized(target_size)


def build_edit():
    processed = []

    for i, (filename, start, end) in enumerate(CLIPS):
        if not os.path.exists(filename):
            print(f"[!] File not found: {filename} — skipping")
            continue

        print(f"[{i+1}/{len(CLIPS)}] Loading {filename} ({start}s → {end}s)")
        clip = VideoFileClip(filename).subclipped(start, end)
        clip = crop_to_tiktok(clip)

        # Apply text overlays that fall within this clip's time range
        # (relative to output timeline — adjust if needed)
        processed.append(clip)

    if not processed:
        print("[!] No clips loaded. Check your filenames.")
        return

    print("Concatenating clips...")
    final = concatenate_videoclips(processed, method="compose")

    # Add text overlays
    if TEXT_OVERLAYS:
        layers = [final]
        for text, size, color, pos, t_start, t_end in TEXT_OVERLAYS:
            txt = (
                TextClip(
                    text=text,
                    font_size=size,
                    color=color,
                    font="Arial",
                    stroke_color="black",
                    stroke_width=2,
                    method="caption",
                    size=(TIKTOK_SIZE[0] - 80, None),
                )
                .with_start(t_start)
                .with_end(t_end)
                .with_position(pos)
            )
            layers.append(txt)
        final = CompositeVideoClip(layers)

    # Add background music
    if MUSIC_FILE and os.path.exists(MUSIC_FILE):
        print(f"Adding music: {MUSIC_FILE}")
        music = AudioFileClip(MUSIC_FILE).with_volume_scaled(MUSIC_VOLUME)
        # Loop or trim music to match video length
        if music.duration < final.duration:
            loops = int(final.duration / music.duration) + 1
            music = concatenate_audioclips([music] * loops)
        music = music.subclipped(0, final.duration)
        final = final.with_audio(music)

    print(f"Exporting → {OUTPUT_FILE}")
    final.write_videofile(
        OUTPUT_FILE,
        fps=30,
        codec="libx264",
        audio_codec="aac",
        preset="fast",
    )
    print(f"Done! File saved: {OUTPUT_FILE}")


if __name__ == "__main__":
    build_edit()
