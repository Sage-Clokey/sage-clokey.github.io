"""
config.py — Simulation parameters for coral morphogenesis.

All tunable values live here. Tweak these to change growth behavior
without touching simulation logic.
"""

from dataclasses import dataclass, field


@dataclass
class CoralConfig:
    """
    Central parameter store for the coral growth simulation.

    Biological analogy:
        - step_size       : how far a tip extends per iteration (polyp extension rate)
        - branch_prob     : probability a tip spawns a new branch per step (budding rate)
        - turn_noise      : magnitude of random directional wobble (flow turbulence)
        - upward_bias     : strength of phototropic / gravitropic upward pull
        - min_separation  : minimum allowed distance between nodes (physical exclusion)
        - max_nodes       : growth stops when this many nodes exist (energy budget)
        - max_tips        : cap on simultaneous active tips to prevent explosion
        - random_seed     : seed for reproducibility (None = random each run)
        - branching_angle : half-angle in radians for new branch offset from parent dir
        - radius_base     : tube radius at root (skeleton thickness)
        - radius_taper    : how quickly radius decreases with depth (thinning towards tips)
    """

    # Growth kinematics
    step_size: float = 0.4
    turn_noise: float = 0.25
    upward_bias: float = 0.35

    # Branching
    branch_prob: float = 0.06
    branching_angle: float = 0.55          # radians ~31°
    max_tips: int = 60

    # Collision / termination
    min_separation: float = 0.5
    max_nodes: int = 1200

    # Visual geometry
    radius_base: float = 0.18
    radius_taper: float = 0.012            # radius = base - taper * depth

    # Reproducibility
    random_seed: int | None = 42
