Genome Nucleotide Analyzer

A Python tool for analyzing nucleotide and codon composition of genome sequences from FASTA files.

Features

  • Calculates sequence length and GC content percentage
  • Computes codon usage frequencies across the entire genome
  • Compares nucleotide composition between two genomes
  • Accepts any FASTA format input

Usage

python genomeAnalyzer.py
# Prompts for FASTA filename

python compareGenomes.py
# Compares two FASTA files

Output

<filename> analysis

sequence length = X.X Mb
GC content = XX.X%

UAA : X.XXX
UAG : X.XXX
UGA : X.XXX
...

Codons are reported as RNA codons sorted alphabetically, showing relative frequency within each amino acid group.

Files

  • genomeAnalyzer.py — Main analyzer: sequence length, GC%, codon usage
  • compareGenomes.py — Compares nucleotide composition between two genomes
  • sequenceAnalysis.py — NucParams, ProteinParam, FastAreader shared classes

import os
from sequenceAnalysis import FastAreader, NucParams

def compare_genomes(genome1_file, genome2_file):
    """
    Compares the GC content, nucleotide count, amino acid composition, and relative codon bias of two genomes.

    Parameters:
        genome1_file (str): File path of the first genome.
        genome2_file (str): File path of the second genome.

    Returns:
        None

    Prints:
        Comparison results of the two genomes.
    """
    # Load and analyze data from the first genome
    genome1 = load_genome(genome1_file)
    length1_mb = calculate_genome_length_mb(genome1)
    gc_content1 = genome1.countGC()
    nucleotide_count1 = genome1.nucComposition()
    aa_composition1 = genome1.aaComposition()
    codon_bias1 = calculate_codon_bias(genome1)

    # Load and analyze data from the second genome
    genome2 = load_genome(genome2_file)
    length2_mb = calculate_genome_length_mb(genome2)
    gc_content2 = genome2.countGC()
    nucleotide_count2 = genome2.nucComposition()
    aa_composition2 = genome2.aaComposition()
    codon_bias2 = calculate_codon_bias(genome2)

    # Print comparison results
    print("Comparison Results:")
    print("-" * 20)
    print("Genome Length (Mb):")
    print(f"Genome 1: {length1_mb:.2f} Mb")
    print(f"Genome 2: {length2_mb:.2f} Mb")
    print("\nGC Content:")
    print(f"Genome 1: {gc_content1:.2f}%")
    print(f"Genome 2: {gc_content2:.2f}%")
    print("\nNucleotide Count:")
    print("Genome 1:")
    print_nucleotide_count(nucleotide_count1)
    print("\nGenome 2:")
    print_nucleotide_count(nucleotide_count2)
    print("\nAmino Acid Composition:")
    print("Genome 1:")
    print_aa_composition(aa_composition1)
    print("\nGenome 2:")
    print_aa_composition(aa_composition2)
    print("\nRelative Codon Bias:")
    print("Genome 1:")
    print_codon_bias(codon_bias1)
    print("\nGenome 2:")
    print_codon_bias(codon_bias2)

def calculate_genome_length_mb(genome):
    """
    Calculate the length of a genome in megabases (Mb).

    Parameters:
        genome (NucParams): Nucleotide parameters object containing analyzed data from the genome.

    Returns:
        float: Length of the genome in megabases (Mb).
    """
    total_bases = sum(genome.nucComp.values())
    length_mb = total_bases / 10**6
    return length_mb
    
def load_genome(genome_file):
    """
    Load and analyze data from a genome file.

    Parameters:
        genome_file (str): File path of the genome.

    Returns:
        NucParams: Nucleotide parameters object containing analyzed data from the genome.
    """
    # Get the absolute path of the genome file
    genome_path = os.path.abspath(genome_file)

    # Initialize FASTA reader and NucParams object
    myReader = FastAreader(fname=genome_path)
    myNuc = NucParams()

    # Read sequences from genome file and add them to the NucParams object
    for head, seq in myReader.readFasta():
        myNuc.addSequence(seq)

    return myNuc

def calculate_codon_bias(genome):
    """
    Calculate the relative codon bias of a genome.

    Parameters:
        genome (NucParams): Nucleotide parameters object containing analyzed data from the genome.

    Returns:
        dict: Relative codon bias dictionary.
    """
    # Calculate relative codon usage for each codon
    codon_bias = {}
    for codon in genome.codonComp:
        aa = genome.rnaCodonTable[codon]
        if genome.aaComp[aa] != 0:
            val = genome.codonComp[codon] / genome.aaComp[aa]
        else:
            val = 0
        codon_bias[codon] = val

    return codon_bias

def print_nucleotide_count(nucleotide_count):
    """
    Print nucleotide count.

    Parameters:
        nucleotide_count (dict): Nucleotide count dictionary.
    """
    for base, count in nucleotide_count.items():
        print(f"{base}: {count}")

def print_aa_composition(aa_composition):
    """
    Print amino acid composition.

    Parameters:
        aa_composition (dict): Amino acid composition dictionary.
    """
    for aa, count in aa_composition.items():
        print(f"{aa}: {count}")

def print_codon_bias(codon_bias):
    """
    Print relative codon bias.

    Parameters:
        codon_bias (dict): Relative codon bias dictionary.
    """
    for codon, bias in codon_bias.items():
        print(f"{codon}: {bias:.2f}")

if __name__ == "__main__":
    # Provide file paths of the two genomes to compare
    genome1_file = input("Enter file path of genome 1: ")
    genome2_file = input("Enter file path of genome 2: ")

    # Compare the genomes
    compare_genomes(genome1_file, genome2_file)

    myNuc = NucParams()
    for head, seq in myReader.readFasta() :
        myNuc.addSequence(seq)
    # Your existing code to read the FASTA file and analyze sequences
    
    # Calculate the total number of bases
    total_bases = sum(myNuc.nucComp.values())

    # Print the result with two digits after the decimal and label it as Mb

    # Add sequences to the nuc_params object using addSequence method

        else:

        print ('{:s} : {:s} {:5.1f} ({:6d})'.format(codon, aa, val*100, myNuc.codonComp[codon]))
