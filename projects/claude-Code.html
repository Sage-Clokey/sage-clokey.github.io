function useScrollFadeIn() {
  useEffect(() => {
    const els = document.querySelectorAll(".fade-in");
    const obs = new IntersectionObserver(
      (entries) => entries.forEach((e) => { if (e.isIntersecting) e.target.classList.add("visible"); }),
      { threshold: 0.12 }
    );
    els.forEach((el) => obs.observe(el));
    return () => obs.disconnect();
  }, []);
}

# ─── Core Data Types ──────────────────────────────────────────────────────────

@dataclass
class Conditions:
    """

    Life will grow what wants to grow there.

    ----------
    nutrient_sources  : List of (x, y, strength) in normalized [0,1] coords.

    structural_stress : 0.0 = calm environment → dense, smooth growth.

    growth_budget     : Total growth energy available to the system.
    variation_seed    : None = different every run (true life).
                        Integer = reproducible emergence.
    """
    nutrient_sources: List[Tuple[float, float, float]] = field(default_factory=list)
    light_gradient: str = "top"
    structural_stress: float = 0.5
    moisture: float = 0.5
    growth_budget: int = 400
    variation_seed: Optional[int] = None


@dataclass
class GrowthNode:
    """
    A living growth point.

    A node does not follow a blueprint.
    It reads its local environment and decides where to go next.
    Every node is equal — no hierarchy, no central planner.
    """
    x: float
    y: float
    angle: float       # current growth direction in degrees
    energy: float      # resource energy this node carries
    depth: int = 0     # branching depth (0 = trunk)
    age: int = 0
    alive: bool = True
    parent: Optional['GrowthNode'] = None


class EnvironmentField:
    """
    A continuous 2D field of an environmental variable.

    Nutrients, light, moisture — these are the grammar of life.
    Growth nodes read this field rather than following instructions.
    """

    def __init__(self, width: int, height: int):
        self.width = width
        self.height = height
        self.grid = np.zeros((height, width))

    def add_source(self, x: float, y: float, strength: float, radius: float = 0.15):
        """Add a point source that diffuses outward."""
        cx = int(x * self.width)
        cy = int(y * self.height)
        r = int(radius * min(self.width, self.height))
        Y, X = np.ogrid[:self.height, :self.width]
        dist = np.sqrt((X - cx) ** 2 + (Y - cy) ** 2)
        self.grid += strength * np.exp(-dist / (r + 1e-6))

    def set_gradient(self, direction: str):
        """Set a broad directional gradient."""
        Y, X = np.mgrid[0:self.height, 0:self.width]
        if direction == "top":
            self.grid = 1.0 - Y / self.height
        elif direction == "bottom":
            self.grid = Y / self.height
        elif direction == "radial":
            cx, cy = self.width / 2, self.height / 2
            dist = np.sqrt((X - cx) ** 2 + (Y - cy) ** 2)
            self.grid = 1.0 - dist / (max(cx, cy) + 1e-6)
            self.grid = np.clip(self.grid, 0, 1)
        elif direction == "directional":
            # Light from upper-left
            self.grid = (1.0 - Y / self.height) * 0.7 + (1.0 - X / self.width) * 0.3
        else:  # uniform
            self.grid = np.ones((self.height, self.width)) * 0.5

    def smooth(self, sigma: float = 4.0):
        """Diffuse the field — real nutrients and light don't have sharp edges."""
        self.grid = gaussian_filter(self.grid, sigma=sigma)
        if self.grid.max() > 0:
            self.grid /= self.grid.max()

    def sample(self, x: float, y: float) -> float:
        """Read field value at normalized (0–1) coordinates."""
        px = int(np.clip(x * self.width, 0, self.width - 1))
        py = int(np.clip(y * self.height, 0, self.height - 1))
        return float(self.grid[py, px])


# ─── The Living System ────────────────────────────────────────────────────────

class LivingSystem:
    """
    A decentralized living system that grows structure from conditions.

    There is no central planner.
    Each node reads its local environment and responds.
    The structure that emerges is the sum of ten thousand small decisions —
    each one local, each one voluntary, each one alive.

    This is not assembly. This is emergence.
    """

    # Nature's numbers — Fibonacci / golden ratio
    GOLDEN_ANGLE = 137.508   # degrees: the angle that maximizes packing
    PHI = (1 + 5 ** 0.5) / 2  # the golden ratio

    def __init__(self, conditions: Conditions, width: int = 400, height: int = 400):
        self.conditions = conditions
        self.width = width
        self.height = height

        if conditions.variation_seed is not None:
            random.seed(conditions.variation_seed)
            np.random.seed(conditions.variation_seed)

        # Build environmental fields from conditions
        self.nutrient_field = EnvironmentField(width, height)
        self.light_field = EnvironmentField(width, height)
        self._build_environment()

        # Growth state
        self.nodes: List[GrowthNode] = []
        self.segments: List[Tuple[float, float, float, float, int]] = []  # x1,y1,x2,y2,depth
        self.total_grown = 0

        # Seed the initial growth node
        self._seed_growth()

    def _build_environment(self):
        """Translate Conditions into spatial environmental fields."""
        self.light_field.set_gradient(self.conditions.light_gradient)
        self.light_field.smooth(sigma=6)

        if self.conditions.nutrient_sources:
            for nx, ny, strength in self.conditions.nutrient_sources:
                self.nutrient_field.add_source(nx, ny, strength)
            self.nutrient_field.smooth(sigma=10)
        else:
            # Default: soil nutrients concentrate near ground
            self.nutrient_field.set_gradient("bottom")
            self.nutrient_field.smooth(sigma=6)

    def _seed_growth(self):
        """Plant the initial seed at the base."""
        seed = GrowthNode(
            x=0.5,
            y=0.92,
            angle=90.0,  # starts growing upward
            energy=1.0,
            depth=0,
        )
        self.nodes.append(seed)

    def _compute_growth_direction(self, node: GrowthNode) -> float:
        """
        The core of life-based design.

        The node doesn't follow a blueprint. It probes its surroundings,
        reads the environmental signals, and grows toward what is good.
        This is tropism — life's original intelligence.
        """
        best_angle = node.angle
        best_value = -np.inf

        # Probe a cone of possible directions ahead
        stress = self.conditions.structural_stress
        spread = 50 + stress * 70  # high stress = wider search cone

        for delta in np.linspace(-spread, spread, 15):
            probe_angle = node.angle + delta
            rad = np.deg2rad(probe_angle)

            probe_x = node.x + 0.035 * np.cos(rad)
            probe_y = node.y - 0.035 * np.sin(rad)

            probe_x = np.clip(probe_x, 0, 1)
            probe_y = np.clip(probe_y, 0, 1)

            # Read the environment at this probe point
            light = self.light_field.sample(probe_x, probe_y)
            nutrients = self.nutrient_field.sample(probe_x, probe_y)

            # Environmental signal combines light-seeking and nutrient-seeking
            signal = 0.65 * light + 0.35 * nutrients

            # Gravitropism: slight tendency to grow upward (like real plants)
            upward_bias = 0.08 * np.sin(np.deg2rad(probe_angle))
            signal += upward_bias

            # Thigmotropism: avoid sharp direction changes (inertia of growth)
            angle_cost = abs(delta) / spread * 0.12
            signal -= angle_cost

            # Biological variation — no two growths are identical
            signal += np.random.normal(0, 0.04 + stress * 0.04)

            if signal > best_value:
                best_value = signal
                best_angle = probe_angle

        return best_angle

    def _should_branch(self, node: GrowthNode) -> bool:
        """
        Branching decision — decentralized, local, condition-responsive.

        High stress → more branching (defense, redundancy).
        Low moisture → less branching (conservation).
        Deeper nodes branch less (apical dominance).
        """
        if node.depth >= 9:
            return False

        # Base probability from structural stress
        p = 0.12 * self.conditions.structural_stress

        # Nutrient richness encourages branching
        local_nutrients = self.nutrient_field.sample(node.x, node.y)
        p += 0.08 * local_nutrients

        # Moisture modulates density
        p *= (0.3 + 0.7 * self.conditions.moisture)

        # Apical dominance: deeper = less branching (exponential suppression)
        p *= (1.0 / (1 + node.depth * 0.35))

        # Golden angle makes branching spacing Fibonacci-optimal
        # (more likely after a minimum growth run)
        if node.age < 3:
            p *= 0.3

        return random.random() < p

    def _step_node(self, node: GrowthNode) -> List[GrowthNode]:
        """
        Advance one growth node by one step.
        Returns any new nodes born from branching.
        """
        if not node.alive or node.energy <= 0:
            return []

        # Choose direction from environmental sensing
        node.angle = self._compute_growth_direction(node)
        rad = np.deg2rad(node.angle)

        # Step size decreases with depth and energy (resource limitation)
        step = 0.020 * node.energy * (1.0 / (1 + node.depth * 0.18))

        new_x = node.x + step * np.cos(rad)
        new_y = node.y - step * np.sin(rad)

        # Boundary check — life respects its environment's edges
        if not (0.01 <= new_x <= 0.99 and 0.01 <= new_y <= 0.99):
            node.alive = False
            return []

        # Record this growth segment
        self.segments.append((node.x, node.y, new_x, new_y, node.depth))

        node.x = new_x
        node.y = new_y
        node.age += 1

        # Energy attenuation — resources diminish with distance from root
        decay = 0.975 - node.depth * 0.008
        node.energy *= max(decay, 0.88)

        if node.energy < 0.045:
            node.alive = False
            return []

        # Branching
        new_nodes = []
        if self._should_branch(node):
            # Golden angle branching — Fibonacci packing in space
            # This is why sunflowers and pine cones look the way they do
            base_offset = self.GOLDEN_ANGLE / 2
            spread = 20 + self.conditions.structural_stress * 45
            jitter = random.uniform(-spread * 0.25, spread * 0.25)
            direction = random.choice([-1, 1])

            branch = GrowthNode(
                x=node.x,
                y=node.y,
                angle=node.angle + direction * (base_offset + jitter),
                energy=node.energy * (0.5 + 0.1 * self.conditions.moisture),
                depth=node.depth + 1,
                parent=node,
            )
            # Parent continues with reduced energy (resource split)
            node.energy *= (0.6 + 0.1 * self.conditions.moisture)
            new_nodes.append(branch)

        return new_nodes

    def grow(self) -> bool:
        """
        Advance the simulation by one generation.
        All living nodes act simultaneously — decentralized, parallel.
        Returns True while growth continues.
        """
        if self.total_grown >= self.conditions.growth_budget:
            return False

        active = [n for n in self.nodes if n.alive]
        if not active:
            return False

        new_nodes = []
        for node in active:
            born = self._step_node(node)
            new_nodes.extend(born)
            self.total_grown += 1

        self.nodes.extend(new_nodes)
        return True

    def grow_fully(self):
        """Let the system grow until its energy budget is exhausted."""
        while self.grow():
            pass

    def get_health_score(self) -> Dict:
        """
        Assess the health of the grown structure.

        A healthy living system shows: diversity of form, spatial balance,
        efficient coverage, and natural branching complexity.
        These emerge from conditions — they cannot be commanded.
        """
        if not self.segments:
            return {}

        depths = [s[4] for s in self.segments]
        xs = [s[2] for s in self.segments]
        ys = [s[3] for s in self.segments]

        # Depth diversity: are multiple scales of structure present?
        depth_diversity = len(set(depths)) / 9

        # Spatial balance: is growth centered or lopsided?
        spatial_balance = 1.0 - abs(np.mean(xs) - 0.5) * 2

        # Coverage: what fraction of the design space is inhabited?
        occupied = set((int(x * 25), int(y * 25)) for x, y in zip(xs, ys))
        coverage = min(1.0, len(occupied) / 625)

        # Branching complexity: richness of the branching network
        branch_nodes = len([n for n in self.nodes if n.depth > 0])
        branching_complexity = min(1.0, branch_nodes / 80)

        # Fibonacci alignment: how well does branching follow golden-angle spacing?
        # Approximated by depth distribution evenness
        depth_counts = np.bincount(depths, minlength=9)[1:]
        if depth_counts.sum() > 0:
            depth_probs = depth_counts / depth_counts.sum()
            # Compare to ideal geometric distribution (PHI decay)
            ideal = np.array([self.PHI ** (-i) for i in range(len(depth_probs))])
            ideal /= ideal.sum()
            fibonacci_alignment = 1.0 - float(np.mean(np.abs(depth_probs - ideal)))
        else:
            fibonacci_alignment = 0.0

        return {
            "depth_diversity":       round(depth_diversity, 3),
            "spatial_balance":       round(max(0, spatial_balance), 3),
            "coverage":              round(coverage, 3),
            "branching_complexity":  round(branching_complexity, 3),
            "fibonacci_alignment":   round(max(0, fibonacci_alignment), 3),
            "total_nodes":           len(self.nodes),
            "total_segments":        len(self.segments),
        }

class APICache:
    """Simple file-backed cache to avoid hammering public APIs."""

    def __init__(self, cache_dir: str = ".lwm_cache"):
        self.path = Path(cache_dir)
        self.path.mkdir(exist_ok=True)

    def _key(self, url: str) -> str:
        return hashlib.md5(url.encode()).hexdigest()

    def get(self, url: str) -> Optional[Any]:
        f = self.path / self._key(url)
        if f.exists() and (time.time() - f.stat().st_mtime) < 86400:  # 24h TTL
            try:
                return json.loads(f.read_text())
            except Exception:
                return None
        return None

    def set(self, url: str, data: Any):
        f = self.path / self._key(url)
        f.write_text(json.dumps(data))


_cache = APICache()


def _get(url: str, params: dict = None, timeout: int = 8) -> Optional[Any]:
    """Cached GET with graceful failure."""
    full_url = url + ("?" + "&".join(f"{k}={v}" for k, v in (params or {}).items())
                      if params else "")
    cached = _cache.get(full_url)
    if cached is not None:
        return cached
    try:
        r = requests.get(url, params=params, timeout=timeout,
                         headers={"Accept": "application/json",
                                  "User-Agent": "LivingWorks-Morphogenesis/1.0"})
        if r.status_code == 200:
            data = r.json()
            _cache.set(full_url, data)
            return data
    except Exception:
        pass
    return None


class MyGeneSource:
    """
    MyGene.info — gene annotation and expression summary.
    https://mygene.info/v3/
    No authentication required.
    """
    BASE = "https://mygene.info/v3"

    @classmethod
    def lookup(cls, symbol: str, species: str = "human") -> Optional[Dict]:
        data = _get(f"{cls.BASE}/query", {
            "q": symbol, "species": species,
            "fields": "symbol,name,summary,pathway.kegg,entrezgene,ensembl.gene",
            "size": 1,
        })
        if data and data.get("hits"):
            return data["hits"][0]
        return None

    @classmethod
    def expression_estimate(cls, symbol: str) -> Optional[float]:
        """
        MyGene.info doesn't expose full expression matrices, but EXAC
        pLI scores and pathway connectivity give a proxy for activity.
        Returns a 0–1 normalized expression estimate.
        """
        hit = cls.lookup(symbol)
        if not hit:
            return None
        # Proxy: number of kegg pathways the gene participates in
        # (more pathways = higher developmental activity)
        pathways = hit.get("pathway", {}).get("kegg", [])
        if isinstance(pathways, dict):
            pathways = [pathways]
        n_pathways = len(pathways) if pathways else 0
        # Normalize: 0 pathways → 0.3, 10+ pathways → 0.9
        return min(0.9, 0.3 + n_pathways * 0.06)

# ─── Expression Profile ───────────────────────────────────────────────────────

@dataclass
class ExpressionProfile:
    """
    The expression level of each morphogenetic gene in a given tissue/context.
    Values are 0–1 (normalized relative to peak expression).
    """
    tissue: str
    organism: str
    expressions: Dict[str, float] = field(default_factory=dict)
    source: str = "literature"        # "api" | "literature" | "mixed"
    phenotype_terms: List[str] = field(default_factory=list)
    raw_api_data: Dict = field(default_factory=dict)

    @classmethod
    def from_literature(cls, tissue: str = "shoot_apical_meristem",
                        organism: str = "generic") -> "ExpressionProfile":
        """Build a profile from curated literature defaults."""
        expressions = {
            gene: info["default_expr"]
            for gene, info in MORPHOGENETIC_GENES.items()
        }
        return cls(tissue=tissue, organism=organism,
                   expressions=expressions, source="literature")

    @classmethod
    def from_api(cls, gene_symbols: List[str], tissue: str = "generic",
                 organism: str = "homo_sapiens",
                 verbose: bool = True) -> "ExpressionProfile":
        """
        Fetch expression data from public genome databases.
        Falls back to literature values for any gene where API fails.
        """
        profile = cls.from_literature(tissue, organism)
        profile.source = "mixed"
        raw = {}

        if verbose:
            print(f"  Querying genome databases for {len(gene_symbols)} genes ...")

        for symbol in gene_symbols:
            if symbol not in MORPHOGENETIC_GENES:
                continue

            # Try MyGene.info first (most reliable)
            expr = MyGeneSource.expression_estimate(symbol)
            if expr is not None:
                profile.expressions[symbol] = expr
                raw[symbol] = {"source": "mygene", "value": expr}
                if verbose:
                    print(f"    {symbol:10s}  expr={expr:.3f}  [MyGene.info]")
                time.sleep(0.1)   # be polite to free APIs
                continue

            # Fall back to Ensembl for gene info
            info = EnsemblSource.gene_info(symbol, organism)
            if info:
                # Use biotype and description as proxy for expression
                biotype = info.get("biotype", "")
                if biotype == "protein_coding":
                    expr = MORPHOGENETIC_GENES[symbol]["default_expr"]
                    profile.expressions[symbol] = expr
                    raw[symbol] = {"source": "ensembl_proxy", "value": expr}
                    if verbose:
                        print(f"    {symbol:10s}  expr={expr:.3f}  [Ensembl proxy]")
                time.sleep(0.15)
                continue

            if verbose:
                print(f"    {symbol:10s}  expr={profile.expressions[symbol]:.3f}  "
                      f"[literature fallback]")

        profile.raw_api_data = raw

        if len(raw) == len(gene_symbols):
            profile.source = "api"

        return profile

    def activator_mean(self) -> float:
        vals = [self.expressions[g] for g in MORPHOGENETIC_GENES
                if MORPHOGENETIC_GENES[g]["class"] == "activator"
                and g in self.expressions]
        return float(np.mean(vals)) if vals else 0.5

    def inhibitor_mean(self) -> float:
        vals = [self.expressions[g] for g in MORPHOGENETIC_GENES
                if MORPHOGENETIC_GENES[g]["class"] == "inhibitor"
                and g in self.expressions]
        return float(np.mean(vals)) if vals else 0.5

    def activation_ratio(self) -> float:
        """act / (act + inh) — the master morphogenetic index."""
        a = self.activator_mean()
        i = self.inhibitor_mean()
        return a / (a + i + 1e-9)

    def dominant_pathway(self) -> str:
        pathway_totals: Dict[str, float] = {}
        for gene, expr in self.expressions.items():
            if gene in MORPHOGENETIC_GENES:
                pw = MORPHOGENETIC_GENES[gene]["pathway"]
                pathway_totals[pw] = pathway_totals.get(pw, 0) + expr
        if not pathway_totals:
            return "unknown"
        return max(pathway_totals, key=pathway_totals.get)


# ─── Turing Reaction-Diffusion Field ─────────────────────────────────────────

@dataclass
class TuringParameters:
    """
    Gray-Scott reaction-diffusion parameters derived from gene expression.

    Gray-Scott model:
        du/dt = Du·∇²u  −  u·v²  +  f·(1−u)
        dv/dt = Dv·∇²v  +  u·v²  −  (f+k)·v

    where:
        u = activator morphogen  (e.g. WNT, FGF)
        v = inhibitor morphogen  (e.g. DKK, BMP)
        Du, Dv = diffusion coefficients (Du > Dv for Turing instability)
        f  = feed rate  (activator production, driven by activator genes)
        k  = kill rate  (inhibitor removal, driven by inhibitor genes)

    Different (f, k) regions produce different morphologies:
        f=0.037, k=0.060  →  spots   (leopard)
        f=0.060, k=0.062  →  stripes (zebra)
        f=0.025, k=0.055  →  coral / branching
        f=0.039, k=0.058  →  holes / inverse spots
    """
    Du: float = 0.16    # activator diffusivity
    Dv: float = 0.08    # inhibitor diffusivity  (must be < Du for Turing instability)
    f:  float = 0.037   # feed rate
    k:  float = 0.060   # kill rate
    noise: float = 0.02

    @classmethod
    def from_expression(cls, profile: ExpressionProfile) -> "TuringParameters":
        """
        Map gene expression levels to Turing parameters.

        The key insight: the ratio of activator:inhibitor expression controls
        which morphological regime we're in — this is nature's parameterization.
        """
        # Activator diffusion: faster in high-FGF / high-WNT tissues
        fgf_expr  = profile.expressions.get("FGF8",  0.65)
        wnt_expr  = profile.expressions.get("WNT5A", 0.72)
        Du = 0.12 + 0.06 * (0.6 * fgf_expr + 0.4 * wnt_expr)

        # Inhibitor diffusion: modulated by BMP and NOTCH
        bmp_expr    = profile.expressions.get("BMP4",   0.45)
        notch_expr  = profile.expressions.get("NOTCH1", 0.60)
        Dv = 0.05 + 0.04 * (0.5 * bmp_expr + 0.5 * notch_expr)
        Dv = min(Dv, Du * 0.65)   # maintain Turing instability condition

        # Feed rate (f): how fast the activator is produced
        # High WNT3A + MYC = high f = more activator production
        myc_expr  = profile.expressions.get("MYC",  0.70)
        wnt3_expr = profile.expressions.get("WNT3A", 0.55)
        f = 0.020 + 0.028 * (0.5 * myc_expr + 0.5 * wnt3_expr)

        # Kill rate (k): how fast the inhibitor is cleared
        # High DKK1 + SFRP1 = more inhibitor → higher k
        dkk_expr  = profile.expressions.get("DKK1",  0.38)
        sfrp_expr = profile.expressions.get("SFRP1", 0.42)
        tp53_expr = profile.expressions.get("TP53",  0.50)
        k = 0.045 + 0.022 * (0.4 * dkk_expr + 0.3 * sfrp_expr + 0.3 * tp53_expr)

        # Noise: modulated by SHH (cell polarity variation)
        shh_expr = profile.expressions.get("SHH", 0.58)
        noise = 0.008 + 0.025 * shh_expr

        return cls(Du=Du, Dv=Dv, f=f, k=k, noise=noise)

    def morphology_label(self) -> str:
        """Classify the expected morphological pattern from f,k values."""
        f, k = self.f, self.k
        if f < 0.030 and k < 0.058:
            return "coral / branching"
        elif f < 0.040 and k < 0.062:
            return "spots"
        elif f >= 0.050 and k > 0.058:
            return "stripes"
        elif f >= 0.038 and k < 0.062:
            return "holes / inverse-spots"
        elif f < 0.030 and k >= 0.058:
            return "worm-like / labyrinthine"
        else:
            return "mixed / transitional"

    def __str__(self) -> str:
        return (f"TuringParameters(Du={self.Du:.4f}, Dv={self.Dv:.4f}, "
                f"f={self.f:.4f}, k={self.k:.4f})  →  {self.morphology_label()}")


class TuringField:
    """
    Gray-Scott reaction-diffusion simulation.

    Produces the genetic pre-pattern: a spatial field of activator
    concentration that encodes where growth should and should not happen.
    This is what genes actually do in morphogenesis — they don't draw
    shapes, they set up chemical gradients that cells respond to.
    """

    def __init__(self, width: int, height: int, params: TuringParameters,
                 seed: Optional[int] = None):
        self.width = width
        self.height = height
        self.params = params

        rng = np.random.default_rng(seed)

        # Initialize: near-uniform + small noise (mimics undifferentiated tissue)
        self.u = np.ones((height, width)) * 0.5 + rng.normal(0, params.noise, (height, width))
        self.v = np.ones((height, width)) * 0.25 + rng.normal(0, params.noise * 0.5, (height, width))

        # Seed: small patch of high-v to break symmetry (like an organizer region)
        cx, cy = width // 2, height // 2
        r = max(5, width // 15)
        Y, X = np.ogrid[:height, :width]
        mask = (X - cx) ** 2 + (Y - cy) ** 2 < r ** 2
        self.u[mask] = 0.50
        self.v[mask] = 0.25

        self.generation = 0

    def step(self, n: int = 1):
        """Advance the reaction-diffusion system by n time steps."""
        Du, Dv, f, k = self.params.Du, self.params.Dv, self.params.f, self.params.k
        dt = 1.0

        for _ in range(n):
            # Laplacian (periodic boundaries)
            lap_u = (
                np.roll(self.u, 1, 0) + np.roll(self.u, -1, 0) +
                np.roll(self.u, 1, 1) + np.roll(self.u, -1, 1) - 4 * self.u
            )
            lap_v = (
                np.roll(self.v, 1, 0) + np.roll(self.v, -1, 0) +
                np.roll(self.v, 1, 1) + np.roll(self.v, -1, 1) - 4 * self.v
            )

            # Reaction term
            uvv = self.u * self.v * self.v

            # Gray-Scott update
            self.u += dt * (Du * lap_u - uvv + f * (1 - self.u))
            self.v += dt * (Dv * lap_v + uvv - (f + k) * self.v)

            # Keep in valid range
            self.u = np.clip(self.u, 0, 1)
            self.v = np.clip(self.v, 0, 1)

            self.generation += 1

    def run_to_pattern(self, steps: int = 4000, verbose: bool = True):
        """Run until the pattern has stabilized."""
        if verbose:
            print(f"  Running Turing reaction-diffusion ({steps} steps) ...")
        chunk = max(1, steps // 20)
        for i in range(0, steps, chunk):
            self.step(chunk)
            if verbose:
                pct = int((i + chunk) / steps * 100)
                bar = "█" * (pct // 5) + "░" * (20 - pct // 5)
                print(f"\r  [{bar}] {pct:3d}%  gen={self.generation}", end="", flush=True)
        if verbose:
            print()

    @property
    def activator_field(self) -> np.ndarray:
        """The activator concentration: this is where growth wants to happen."""
        u = self.u.copy()
        if u.max() > u.min():
            u = (u - u.min()) / (u.max() - u.min())
        return u

    @property
    def inhibitor_field(self) -> np.ndarray:
        """The inhibitor concentration: this is where growth is suppressed."""
        v = self.v.copy()
        if v.max() > v.min():
            v = (v - v.min()) / (v.max() - v.min())
        return v

    @property
    def net_field(self) -> np.ndarray:
        """Net morphogenetic signal: activator minus inhibitor."""
        net = self.activator_field - self.inhibitor_field * 0.6
        net = np.clip(net, 0, 1)
        if net.max() > 0:
            net /= net.max()
        return net


# ─── Genetic Conditions ───────────────────────────────────────────────────────

@dataclass
class GeneticConditions(Conditions):
    """
    Extends Conditions with genetic data.

    The genetic data feeds into the Turing field, which in turn
    becomes the environmental field that growth nodes navigate.
    """
    profile: Optional[ExpressionProfile] = None
    turing_params: Optional[TuringParameters] = None
    turing_steps: int = 4000
    phenotype_label: str = ""

    @classmethod
    def from_expression_profile(cls, profile: ExpressionProfile,
                                 turing_steps: int = 4000) -> "GeneticConditions":
        """Derive all simulation parameters from a gene expression profile."""
        params = TuringParameters.from_expression(profile)
        ar = profile.activation_ratio()

        # Map genetic state → ecological conditions
        # (these are the bridges between molecular and organismal scale)
        structural_stress = 1.0 - ar                     # high inhibition = stress
        moisture = profile.expressions.get("VEGFA", 0.5) # VEGF = vascular supply ≈ moisture
        growth_budget = int(250 + 350 * ar)              # high activation = more growth
        light_gradient = cls._axis_to_gradient(profile)

        # Dominant nutrient sources from gradient gene axes
        nutrient_sources = cls._derive_sources(profile)

        return cls(
            profile=profile,
            turing_params=params,
            turing_steps=turing_steps,
            structural_stress=round(structural_stress, 3),
            moisture=round(moisture, 3),
            growth_budget=growth_budget,
            light_gradient=light_gradient,
            nutrient_sources=nutrient_sources,
            phenotype_label=profile.dominant_pathway(),
        )

    @staticmethod
    def _axis_to_gradient(profile: ExpressionProfile) -> str:
        nodal_expr = profile.expressions.get("NODAL", 0.5)
        wnt8_expr  = profile.expressions.get("WNT8A", 0.48)
        bmp2_expr  = profile.expressions.get("BMP2",  0.60)
        if nodal_expr > 0.6:
            return "directional"
        elif bmp2_expr > 0.65:
            return "top"
        elif wnt8_expr > 0.55:
            return "radial"
        else:
            return "top"

    @staticmethod
    def _derive_sources(profile: ExpressionProfile) -> List[Tuple[float, float, float]]:
        """Gradient genes create nutrient-equivalent sources in simulation space."""
        sources = []
        bmp2  = profile.expressions.get("BMP2",  0.0)
        nodal = profile.expressions.get("NODAL", 0.0)
        wnt8  = profile.expressions.get("WNT8A", 0.0)
        vegfa = profile.expressions.get("VEGFA", 0.0)

        if bmp2 > 0.4:   sources.append((0.5, 0.05, bmp2))       # dorsal signal → top
        if nodal > 0.4:  sources.append((0.15, 0.5, nodal))       # left-side signal
        if wnt8 > 0.4:   sources.append((0.5, 0.92, wnt8))        # posterior → base
        if vegfa > 0.5:  sources.append((0.5, 0.5, vegfa * 0.7))  # radial vascular

        if not sources:
            sources = [(0.5, 0.9, 0.8)]  # default: ground-level resource

        return sources


# ─── Morphogenesis System ─────────────────────────────────────────────────────

class MorphogenesisSystem(LivingSystem):
    """
    Extends LivingSystem with a Turing-derived morphogenetic field.

    Instead of reading a hand-designed gradient, growth nodes navigate
    the landscape produced by real gene expression data — the same
    molecular logic that drives morphogenesis in living organisms.
    """

    def __init__(self, genetic_conditions: GeneticConditions,
                 width: int = 200, height: int = 200,
                 verbose: bool = True):
        self.genetic_conditions = genetic_conditions
        self.verbose = verbose

        # Build the Turing field from genetic data
        if verbose:
            print(f"  Turing parameters: {genetic_conditions.turing_params}")
        self.turing = TuringField(
            width, height,
            genetic_conditions.turing_params,
            seed=genetic_conditions.variation_seed,
        )
        self.turing.run_to_pattern(genetic_conditions.turing_steps, verbose=verbose)

        # Inject Turing field into the base environment
        # This overwrites the simple gradient with the genetic pre-pattern
        super().__init__(genetic_conditions, width=width, height=height)
        self._inject_turing_field()

    def _inject_turing_field(self):
        """
        Replace the hand-crafted environment fields with the
        genetically-derived Turing field.
        """
        net = self.turing.net_field
        act = self.turing.activator_field

        # The nutrient field becomes the activator concentration
        self.nutrient_field.grid = net

        # Light field: blend Turing activator with the gradient axis
        # (top-down light is not genetic, but gravity/solar axis is real)
        axis = self.light_field.grid.copy()
        self.light_field.grid = 0.45 * act + 0.55 * axis
        if self.light_field.grid.max() > 0:
            self.light_field.grid /= self.light_field.grid.max()

    def get_health_score(self) -> Dict:
        health = super().get_health_score()
        # Add genetic context
        if self.genetic_conditions.profile:
            p = self.genetic_conditions.profile
            health["activation_ratio"]    = round(p.activation_ratio(), 3)
            health["dominant_pathway"]    = p.dominant_pathway()
            health["turing_morphology"]   = self.genetic_conditions.turing_params.morphology_label()
        return health

    def __init__(self, genetic_conditions: GeneticConditions,
                 label: str = "", verbose: bool = True):
        self.genetic_conditions = genetic_conditions
        self.label = label
        self.verbose = verbose
        self.system: Optional[MorphogenesisSystem] = None

    @classmethod
    def from_organism(cls, organism_key: str,
                      verbose: bool = True) -> "MorphogenesisSimulation":
        """Load a preset organism profile."""
        if organism_key not in cls.PRESETS:
            raise ValueError(f"Unknown organism '{organism_key}'. "
                             f"Available: {list(cls.PRESETS.keys())}")

        preset = cls.PRESETS[organism_key]
        profile = ExpressionProfile(
            tissue=preset["tissue"],
            organism=preset["organism"],
            expressions=preset["expressions"],
            source="literature",
        )

        gc = GeneticConditions.from_expression_profile(
            profile, turing_steps=preset["turing_steps"]
        )
        gc.variation_seed = preset["seed"]

        if verbose:
            print(f"\n  Organism:  {preset['label']}")
            print(f"  Note:      {preset['note']}")
            print(f"  act_ratio: {profile.activation_ratio():.3f}  |  "
                  f"dominant: {profile.dominant_pathway()}")
            print()

        return cls(gc, label=preset["label"], verbose=verbose)

    @classmethod
    def from_gene_query(cls, gene_symbols: List[str],
                        tissue: str = "generic",
                        organism: str = "homo_sapiens",
                        turing_steps: int = 4000,
                        verbose: bool = True) -> "MorphogenesisSimulation":
        """
        Fetch live data from genome databases for the given genes
        and run the simulation.
        """
        if verbose:
            print(f"\n  Fetching genetic data from public genome databases ...")
            print(f"  Genes: {', '.join(gene_symbols)}")

        profile = ExpressionProfile.from_api(
            gene_symbols, tissue=tissue, organism=organism, verbose=verbose
        )
        gc = GeneticConditions.from_expression_profile(profile, turing_steps=turing_steps)

        return cls(gc, label=f"{tissue} ({organism})", verbose=verbose)

    def run(self, grid_size: int = 200) -> "MorphogenesisSimulation":
        """Build the Turing field, grow the structure, and return self."""
        if self.verbose:
            print(f"  Building morphogenetic system  (grid={grid_size}×{grid_size}) ...")

        self.system = MorphogenesisSystem(
            self.genetic_conditions,
            width=grid_size, height=grid_size,
            verbose=self.verbose,
        )

        if self.verbose:
            print(f"  Growing structure ...")
        self.system.grow_fully()

        health = self.system.get_health_score()
        if self.verbose:
            print()
            print(f"  ── Result ───────────────────────────────────────────")
            print(f"  segments        {health.get('total_segments', 0)}")
            print(f"  nodes           {health.get('total_nodes', 0)}")
            print(f"  depth diversity {health.get('depth_diversity', 0):.3f}")
            print(f"  spatial balance {health.get('spatial_balance', 0):.3f}")
            print(f"  coverage        {health.get('coverage', 0):.3f}")
            print(f"  fib alignment   {health.get('fibonacci_alignment', 0):.3f}")
            print(f"  activation ratio  {health.get('activation_ratio', '?')}")
            print(f"  turing morphology {health.get('turing_morphology', '?')}")

        return self

    def visualize(self, title: str = ""):
        """Render the full 5-panel morphogenesis visualization."""
        if self.system is None:
            raise RuntimeError("Call .run() before .visualize()")
        t = title or f"LivingWorks Morphogenesis — {self.label}"
        MorphogenesisVisualizer(self.system).render(t)
