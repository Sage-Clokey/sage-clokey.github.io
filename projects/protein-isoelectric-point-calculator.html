Protein Isoelectric Point Calculator

A Python tool that calculates the isoelectric point (pI) and net charge of proteins from amino acid sequences.

What It Does

The isoelectric point (pI) is the pH at which a protein carries no net electrical charge — critical for protein purification, gel electrophoresis, and understanding protein behavior in solution.

The calculator: - Computes net charge from ionizable residues (D, E, C, Y, H, K, R) - Uses experimentally determined pKa values for each residue type - Accounts for N- and C-terminal charges - Calculates pI by finding the pH where net charge = 0

Usage

python pIsoelectricPoint.py

Or import the class directly:

from ProteinIsoelectricPointCalculator import ProteinIsoelectricPointCalculator

calc = ProteinIsoelectricPointCalculator()

class ProteinIsoelectricPointCalculator:
    def __init__(self):
        # Dictionary of pKa values for the amino acids
        self.pKa_values = {
            'D': 3.90, 'E': 4.07, 'C': 8.18, 'Y': 10.46,
            'H': 6.00, 'K': 10.54, 'R': 12.48
        }

    def calculate_charge(self, sequence):
        # Initialize total charge
        total_charge = 0

        # Count the occurrence of each amino acid
        aa_counts = {aa: sequence.count(aa) for aa in self.pKa_values.keys()}

        # Calculate charge contributed by acidic residues
        total_charge += sum(aa_counts[aa] * (-10 ** self.pKa_values[aa] / (10 ** self.pKa_values[aa] + 10 ** 3.90)) for aa in ['D', 'E'])

        # Calculate charge contributed by basic residues
        total_charge += sum(aa_counts[aa] * (10 ** 10.54 / (10 ** self.pKa_values[aa] + 10 ** 10.54)) for aa in ['K', 'R'])

        # Calculate charge contributed by cysteine residues (taking into account thiol/disulfide equilibrium)
        total_charge += aa_counts['C'] * (10 ** 10.54 / (10 ** self.pKa_values['C'] + 10 ** 8.18)) / 2

        # Calculate charge contributed by the terminal amino and carboxyl groups
        total_charge += (10 ** 10.54 / (10 ** 3.1 + 10 ** 10.54)) - (10 ** 3.9 / (10 ** self.pKa_values['C'] + 10 ** 3.9))

        return total_charge

    def calculate_pI(self, sequence):
        total_charge = self.calculate_charge(sequence)
        # Calculate pI
        pI = total_charge / len(sequence)

        return pI

# Example usage:
protein_calculator = ProteinIsoelectricPointCalculator()
protein_sequence = "DEYHKRCY"
charge = protein_calculator.calculate_charge(protein_sequence)
print("Net charge of the protein:", charge)

pI = protein_calculator.calculate_pI(protein_sequence)
print("Isoelectric point (pI) of the protein:", pI)

class aaparam:
    aa2chargePos = {'K': 10.5, 'R': 12.4, 'H': 6}
    aa2chargeNeg = {'D': 3.86, 'E': 4.25, 'C': 8.33, 'Y': 10}

    def __init__(self, protein):
        self.protein = protein
        self.aaComposition = self.calculateAaComposition()

    def calculateAaComposition(self):
        # Calculate the amino acid composition and store it in the composition dictionary
        composition = {aa: 0 for aa in self.protein}

        for aa in self.protein:
            if aa.upper() in composition:
                composition[aa.upper()] += 1

        return composition

    def charge(self, pH):
        # Initialize total charge
        total_charge = 0

        # Count the occurrence of each amino acid
        aa_counts = self.aaComposition

        # Calculate positive charges
        total_charge += sum(aa_counts.get(aa, 0) * (self.aa2chargePos.get(aa, 0) / (1 + 10 ** (self.aa2chargePos.get(aa, 0) - pH))) for aa in self.aa2chargePos.keys())

        # Calculate negative charges (Note: Reverse calculation for negative amino acids)
        total_charge -= sum(aa_counts.get(aa, 0) * (self.aa2chargeNeg.get(aa, 0) / (1 + 10 ** (pH - self.aa2chargeNeg.get(aa, 0)))) for aa in self.aa2chargeNeg.keys())

        return total_charge

    def pI(self):
        # Binary search to find pH at which net charge is closest to zero
        def find_pH():
            lower_pH, upper_pH = 0.0, 14.0
            precision = 0.01

            while upper_pH - lower_pH > precision:
                mid_pH = (lower_pH + upper_pH) / 2
                charge_at_mid_pH = self.charge(mid_pH)

                if charge_at_mid_pH < 0:
                    lower_pH = mid_pH
                else:
                    upper_pH = mid_pH

            return (lower_pH + upper_pH) / 2

        return find_pH()

# Example usage:
protein_sequence = "DEYHKRCY"
param_calculator = aaparam(protein_sequence)
pI = param_calculator.pI()
print("Isoelectric point (pI) of the protein:", pI)
