Implementations

ArrayList (arraylist/)

A generic growable array list implementing the List<AnyType> interface.

  • Extends AbstractCollection<AnyType>
  • Constructor accepts a Collection and pre-sizes to match it
  • Supports standard list operations: add, remove, get, set, size
  • Generic type bound: AnyType extends Comparable<? super AnyType>

AVL Tree (avl-tree/)

A self-balancing binary search tree (AVL Tree) implementation.

  • Automatic rebalancing via rotations on insert/delete
  • insert, remove, contains, findMin, findMax operations
  • Result wrapper class for return values
  • UnderflowException for operations on empty trees
  • O(log n) guaranteed for all operations

Requirements

  • Java 8+
  • No external dependencies

    public static <AnyType extends Comparable<? super AnyType>> int binSearch(A1232SClo<AnyType> list, AnyType item) {
        int low = 0;
        int high = list.size() - 1;
        while (low <= high) {
            int mid = (low + high) / 2;
            int cmp = 1;
            if(list.get(mid) != null){
                cmp = compare(item, list.get(mid));
            }


            if (cmp < 0)
                high = mid - 1;
            else if (cmp > 0)
                low = mid + 1;
            else
                return mid;
        }
        return -1; // Not found
    }

    private A1233SCloAvlNode<AnyType> remove( AnyType x, A1233SCloAvlNode<AnyType> t )
    {
        if( t == null )
            return t;   // Item not found; do nothing

        int compareResult = x.compareTo( t.element );

        if( compareResult < 0 )
            t.left = remove( x, t.left );
        else if( compareResult > 0 )
            t.right = remove( x, t.right );
        else if( t.left != null && t.right != null ) // Two children

    /**
     * Rotate binary tree node with left child.
     * For AVL trees, this is a single rotation for case 1.
     * Update heights, then return new root.
     */
    private A1233SCloAvlNode<AnyType> rotateWithLeftChild( A1233SCloAvlNode<AnyType> k2 )
    {
        A1233SCloAvlNode<AnyType> k1 = k2.left;
        k2.left = k1.right;
        k1.right = k2;
        k2.height = Math.max( height( k2.left ), height( k2.right ) ) + 1;
        k1.height = Math.max( height( k1.left ), k2.height ) + 1;
        return k1;
    }

    /**
     * Rotate binary tree node with right child.
     * For AVL trees, this is a single rotation for case 4.
     * Update heights, then return new root.
     */
    private A1233SCloAvlNode<AnyType> rotateWithRightChild( A1233SCloAvlNode<AnyType> k1 )
    {
        A1233SCloAvlNode<AnyType> k2 = k1.right;
        k1.right = k2.left;
        k2.left = k1;
        k1.height = Math.max( height( k1.left ), height( k1.right ) ) + 1;
        k2.height = Math.max( height( k2.right ), k1.height ) + 1;
        return k2;
    }

    /**
     * Double rotate binary tree node: first left child
     * with its right child; then node k3 with new left child.
     * For AVL trees, this is a double rotation for case 2.
     * Update heights, then return new root.
     */
    private A1233SCloAvlNode<AnyType> doubleWithLeftChild( A1233SCloAvlNode<AnyType> k3 )
    {
        k3.left = rotateWithRightChild( k3.left );
        return rotateWithLeftChild( k3 );
    }

    /**
     * Double rotate binary tree node: first right child
     * with its left child; then node k1 with new right child.
     * For AVL trees, this is a double rotation for case 3.
     * Update heights, then return new root.
     */
    private A1233SCloAvlNode<AnyType> doubleWithRightChild( A1233SCloAvlNode<AnyType> k1 )
    {
        k1.right = rotateWithLeftChild( k1.right );
        return rotateWithRightChild( k1 );
    }

    private void getMode(A1233SCloAvlNode<AnyType> t){

        if (t != null) {
            getMode(t.left);


            checkCount = 0;

                for (AnyType obj : t.objectsWithSameKey) {
                    checkCount += 1;
                }

            //System.out.print( "\n   Data: " + t.element + "  | Count:"+ checkCount + "   Status: "  + status(t));
            if (checkCount > modeCount && !t.deleted ) {

            }
            getMode(t.right);
        }

    }
