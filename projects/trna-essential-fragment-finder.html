README

tRNA Essential Fragment Finder

A Python tool that identifies the minimal essential unique fragments of tRNA sequences — the shortest subsequence that uniquely identifies each tRNA among a set.

What It Does

Given a FASTA file of tRNA sequences, for each tRNA the tool finds every fragment (substring) that: 1. Appears in this tRNA but not in any other tRNA in the set 2. Is the shortest such fragment (essential unique fragment)

This is useful for designing tRNA-specific probes, primers, or diagnostic sequences.

Algorithm

Uses a power-set approach: - Generates all possible substrings of each tRNA sequence - Builds an "others" set of all substrings from all other tRNAs - Identifies substrings unique to the target tRNA - Filters to the shortest (essential) ones

Usage

python findUnique.py bos-tRNA.fa

Output files are written to output_<input>_tRNA_Essentials/ with one file per tRNA.

Files

  • findUnique.py — FindUnique class with power-set unique fragment logic
  • sequenceAnalysis.py — FastAreader FASTA parser
  • bos-tRNA.fa — Full Bos taurus mitochondrial tRNA sequences
  • bos-tRNA-7.fa — Subset of 7 tRNA sequences for testing

Output Format

Each output file contains the essential unique fragments for that tRNA, one per line:

ACGU
CGUA
...

class FindUnique():
    """Class to find essential unique fragments of tRNAs"""
    instantiate = []

    def __init__(self, sequence,head, backgroundFolder, background = False, ):
        """
        Initialize the FindUnique object.

        Args:
            sequence (str): The input sequence.
            head (str): The header for the sequence.
            backgroundFolder (str): Path to the background folder.
            background (bool): Flag to enable background processing.
        """
        self.background = background
        self.head = head.replace(' ', '')
        self.sequence = self.removeAlignmentChars(sequence)

        # Create a directory for the class if it doesn't exist
        if self.background:
            classDirectory = backgroundFolder
            if not os.path.exists(classDirectory):
                os.makedirs(classDirectory)
        
            # Create a directory specific to the class
            self.classDirectory = os.path.join(classDirectory, self.__class__.__name__)
            if not os.path.exists(self.classDirectory):
                os.makedirs(self.classDirectory)
        
        # Create a file specific to the object if background processing is enabled
        
            backgroundFilename = self.sanitizeFilename(f"background_{self.head}.txt")
            self.backgroundFile = os.path.join(self.classDirectory, backgroundFilename)
            with open(self.backgroundFile, 'w'):
                pass
            print(self.backgroundFile)
        else:
            self.backgroundFile = None
        self.pSet = self.powerSet()
        self.others = set()
        self.uniques = set()
        self.nonEssentials = set()
        self.essentials = list()
        bisect.insort_left(FindUnique.instantiate, self, key=lambda x: x.head)

    def sanitizeFilename(self,filename):
        """
        Sanitize a filename by replacing invalid characters.

        Args:
            filename (str): The input filename.

        Returns:
            str: Sanitized filename.
        """
        # Replace invalid characters with underscores
        return re.sub(r'[^\w.-]', '_', filename)
    

    def removeAlignmentChars(self,sequence):
        """
        Remove alignment characters from a sequence.

        Args:
            sequence (str): The input sequence.

        Returns:
            str: Sequence with alignment characters removed.
        """
        alignmentChars = ['.', '_', '-']
        cleanedSequence = ''.join(char for char in sequence if char not in alignmentChars)
        return cleanedSequence

    def initialize(self):
        """Initialize the object by finding unique and essential fragments."""
        self.others = self.buildOthers()
        self.uniques = self.findUniques()
        self.nonEssentials = self.findNonEssentials()
        self.essentials = self.findEssentials()

    def powerSet(self):
        """
        Generate the power set of the sequence.

        Returns:
            set: Power set of the sequence.
        """
        powerSet =  set()
        if self.background:
            self.writeToBackground('')
            self.writeToBackground(self.head)
            self.writeToBackground(24 * "-" + self.sequence)
        for left in range(len(self.sequence)):
            for right in range(left + 1,len(self.sequence) + 1):
                if self.background == True:
                    self.writeToBackground(' left: {:<3}  Right: {:>3}  {}'.format(left, right, ' ' * left + self.sequence[left:right]))
                powerSet.add(self.sequence[left:right])
            if self.background == True:
                self.writeToBackground(24 * "-" + self.sequence)
            
        return powerSet
    
    def buildOthers(self):
        """
        Build the set of all other tRNA sequences.

        Returns:
            set: Set of all other tRNA sequences.
        """
        others = set()
        for object in FindUnique.instantiate:
            if object != self:
                others.update(object.pSet)
        if self.background:
            self.writeToBackground('builtOthers')
            self.writeToBackground(others)
        return others
    
    def findUniques(self):
        """
        Find unique fragments in the sequence.

        Returns:
            set: Set of unique fragments.
        """
        unique = self.pSet.difference(self.others)
        if self.background:
            
            self.writeToBackground('found Uniques')
            self.writeToBackground(unique)
        return unique
    
    def findNonEssentials(self):
        """
        Find non-essential fragments in the sequence.

        Returns:
            set: Set of non-essential fragments.
        """
        nonEssential = set()
        for element in self.uniques:
            if element[1:] in self.uniques or element[:-1] in self.uniques:
                nonEssential.add(element)       
        return nonEssential
    
    def findEssentials(self):
        """
        Find essential fragments in the sequence.

        Returns:
            list: List of essential fragments.
        """
        if self.background:
            self.writeToBackground('found Essential')
        
        essentials = self.uniques.difference(self.nonEssentials)
        #result = [[fragment, index] for fragment in essentials for index in range(len(self.sequence) - len(fragment) + 1) if self.sequence[index:index + len(fragment)] == fragment]
        result = []
        for fragment in essentials:
            index = self.sequence.find(fragment)
            while index != -1:
                result.append([fragment, index])
                index = self.sequence.find(fragment, index + 1)


        result.sort(key=lambda x: x[1])
        if self.background == True:
            self.writeToBackground(result)
        return result
    
    def showEssentials(self):
        """Print essential fragments for the sequence."""
        print('\n essential fragments for: ' + self.head)
        print(self.sequence + '\n')
        for fragment in self.essentials:
            #print('fragments:' + fragment[0])
            #print('index: ' + str(fragment[1]))
            print(fragment[1] * "." + fragment[0])

    def showAllEssentials(self):
        """Print essential fragments for all sequences."""
        for tRNA in FindUnique.instantiate:
            tRNA.showEssentials()

        Args:
            folder (str): Output folder path.

            counter += 1
            folder = f'{baseDirectory}_{counter}'
        print(folder)

        for tRNA in FindUnique.instantiate:
            tRNA.writeEssentialsToFile(folder)

    def writeEssentialsToFile(self,folder):
        """
        Write essential fragments for the sequence to a file.

        Args:
            folder (str): Output folder path.
        """
        # Create a directory with the class name if it doesn't exist
        directory = folder
        if not os.path.exists(directory):
            os.makedirs(directory)
        
        # Sanitize the header to remove invalid characters from the file name
        sanitizedHeader = re.sub(r'[^\w\s.-]', '-', self.head)
        
        # Construct the filename using the sanitized header
        filename = os.path.join(directory, f'{sanitizedHeader}.txt')
        print("    " + filename)
        try:
        # Write essential fragments to the file
            with open(filename, 'w', encoding = 'utf-8') as f:
                f.write(f'Essential fragments for: {self.head}\n')
                f.write(f'{self.sequence}\n\n')
                for fragment in self.essentials:
                    f.write(f'{fragment[1] * "."}{fragment[0]}\n')
        except Exception as e:
            print(f"Error writing to file '{filename}': {e}")
        
    
    def writeToBackground(self, message):
        """
        Write a message to the background file if background processing is enabled.

        Args:
        message (str): The message to be written to the background file.
        """
        if self.background:
            try:
                with open(self.backgroundFile, 'a', encoding = 'utf-8') as f:
                    f.write(str(message) + '\n')
            except UnicodeEncodeError as e:
                print(f"Error writing to background file: {e}")
